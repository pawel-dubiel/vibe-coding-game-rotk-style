112    def _get_max_scroll(self) -> int:
113        """Get maximum scroll offset"""
114        total_height = 10 * (self.slot_height + self.slot_spacing)
115        visible_height = self.max_visible_slots * (self.slot_height + self.slot_spacing)
116        return max(0, total_height - visible_height)

118    def _get_scrollbar_rect(self) -> pygame.Rect:
119        """Calculate scrollbar handle rectangle"""
120        if not self._is_scrollbar_visible():
121            return pygame.Rect(0, 0, 0, 0)
122            
123        # Calculate handle size and position
124        max_scroll = self._get_max_scroll()
125        if max_scroll == 0:
126            handle_height = self.scrollbar_height
127            handle_y = self.scrollbar_y
128        else:
129            visible_ratio = self.list_height / (10 * (self.slot_height + self.slot_spacing))
130            handle_height = int(self.scrollbar_height * visible_ratio)
131            handle_height = max(30, handle_height)  # Minimum handle size
132            
133            scroll_ratio = self.scroll_offset / max_scroll
134            available_space = self.scrollbar_height - handle_height
135            handle_y = self.scrollbar_y + int(available_space * scroll_ratio)
136            
137        return pygame.Rect(self.scrollbar_x, handle_y, self.scrollbar_width, handle_height)

139    def _scroll_to_position(self, y: int):
140        """Scroll to a specific Y position based on scrollbar"""
141        relative_y = y - self.scrollbar_y
142        available_space = self.scrollbar_height - self._get_scrollbar_rect().height
143        
144        if available_space > 0:
145            scroll_ratio = relative_y / available_space
146            max_scroll = self._get_max_scroll()
147            self.scroll_offset = int(max_scroll * scroll_ratio)
148            self.scroll_offset = max(0, min(self.scroll_offset, max_scroll))

166    def handle_event(self, event: pygame.event.Event) -> Optional[dict]:
167        """Handle input events, returns action dict or None"""
168        if not self.visible:
169            return None
170            
171        if event.type == pygame.KEYDOWN:
172            if event.key == pygame.K_ESCAPE:
173                self.hide()
174                return {'action': 'cancel'}
175                
176            # Handle text input for save name
177            if self.input_active and self.action == SaveLoadAction.SAVE:
178                if event.key == pygame.K_BACKSPACE:
179                    self.save_name_input = self.save_name_input[:-1]
180                elif event.key == pygame.K_RETURN:
181                    self.input_active = False
182                elif len(self.save_name_input) < 30:
183                    if event.unicode.isprintable():
184                        self.save_name_input += event.unicode
185                        
186        elif event.type == pygame.MOUSEBUTTONDOWN:
187            mouse_pos = pygame.mouse.get_pos()
188            
189            if event.button == 1:  # Left click
190                # Check slot clicks
191                if self.clip_rect.collidepoint(mouse_pos):
192                    relative_y = mouse_pos[1] - self.list_y + self.scroll_offset
193                    slot_index = relative_y // (self.slot_height + self.slot_spacing)
194                    
195                    if 0 <= slot_index < 10:
196                        self.selected_slot = slot_index + 1
197                        
198                        # For save action, check if slot is occupied
199                        if self.action == SaveLoadAction.SAVE:
200                            slots = self.save_manager.get_save_slots()
201                            if slots[slot_index] is not None:
202                                self.confirm_overwrite = True
203                                
204                # Check scrollbar
205                if self._is_scrollbar_visible():
206                    scrollbar_rect = self._get_scrollbar_rect()
207                    if scrollbar_rect.collidepoint(mouse_pos):
208                        self.dragging_scrollbar = True
209                        self.drag_offset = mouse_pos[1] - scrollbar_rect.y
210                        
211            elif event.button == 4:  # Mouse wheel up
212                self.scroll_offset = max(0, self.scroll_offset - (self.slot_height + self.slot_spacing))
213            elif event.button == 5:  # Mouse wheel down
214                max_scroll = self._get_max_scroll()
215                self.scroll_offset = min(max_scroll, self.scroll_offset + (self.slot_height + self.slot_spacing))
216                    
217            # Check input field click (save mode only)
218            if self.action == SaveLoadAction.SAVE and self.input_rect.collidepoint(mouse_pos):
219                self.input_active = True
220            else:
221                self.input_active = False
222                
223            # Check button clicks
224            if self.button_rects['confirm'].collidepoint(mouse_pos):
225                if self.selected_slot:
226                    if self.confirm_overwrite:
227                        # Confirmed overwrite
228                        return {
229                            'action': 'save',
230                            'slot': self.selected_slot,
231                            'name': self.save_name_input or None,
232                            'overwrite': True
233                        }
234                    elif self.confirm_delete:
235                        # Confirmed delete
236                        result = self.save_manager.delete_save(self.selected_slot)
237                        self.confirm_delete = False
238                        self.selected_slot = None
239                        return {'action': 'deleted', 'result': result}
240                    else:
241                        # Regular action
242                        if self.action == SaveLoadAction.SAVE:
243                            return {
244                                'action': 'save',
245                                'slot': self.selected_slot,
246                                'name': self.save_name_input or None,
247                                'overwrite': False
248                            }
249                        elif self.action == SaveLoadAction.LOAD:
250                            return {
251                                'action': 'load',
252                                'slot': self.selected_slot
253                            }
254                        elif self.action == SaveLoadAction.DELETE:
255                            self.confirm_delete = True
256                            
257            elif self.button_rects['cancel'].collidepoint(mouse_pos):
258                if self.confirm_overwrite:
259                    self.confirm_overwrite = False
260                elif self.confirm_delete:
261                    self.confirm_delete = False
262                else:
263                    self.hide()
264                    return {'action': 'cancel'}
265                    
266        elif event.type == pygame.MOUSEBUTTONUP:
267            if event.button == 1:
268                self.dragging_scrollbar = False
269                
270        elif event.type == pygame.MOUSEMOTION:
271            if self.dragging_scrollbar:
272                mouse_pos = pygame.mouse.get_pos()
273                new_y = mouse_pos[1] - self.drag_offset
274                self._scroll_to_position(new_y)
275                    
276        return None

278    def draw(self):
279        """Draw the menu"""
280        if not self.visible:
281            return
282            
283        # Semi-transparent background
284        overlay = pygame.Surface(self.screen.get_size())
285        overlay.set_alpha(200)
286        overlay.fill(self.colors['background'])
287        self.screen.blit(overlay, (0, 0))
288        
289        # Title
290        title_text = {
291            SaveLoadAction.SAVE: "Save Game",
292            SaveLoadAction.LOAD: "Load Game",
293            SaveLoadAction.DELETE: "Delete Save"
294        }[self.action]
295        
296        title_surface = self.title_font.render(title_text, True, self.colors['text'])
297        title_rect = title_surface.get_rect(center=(self.screen.get_width() // 2, 80))
298        self.screen.blit(title_surface, title_rect)
299        
300        # Set clipping for save slot list
301        self.screen.set_clip(self.clip_rect)
302        
303        # Draw save slots
304        slots = self.save_manager.get_save_slots()
305        mouse_pos = pygame.mouse.get_pos()
306        
307        for i, slot_data in enumerate(slots):
308            slot_num = i + 1
309            
310            # Calculate position with scroll offset
311            slot_y = self.list_y + i * (self.slot_height + self.slot_spacing) - self.scroll_offset
312            
313            # Skip if outside visible area
314            if slot_y + self.slot_height < self.list_y or slot_y > self.list_y + self.list_height:
315                continue
316                
317            rect = pygame.Rect(self.list_x, slot_y, self.slot_width, self.slot_height)
318            
319            # Determine slot color
320            if self.selected_slot == slot_num:
321                color = self.colors['slot_selected']
322            elif rect.collidepoint(mouse_pos) and self.clip_rect.collidepoint(mouse_pos):
323                color = self.colors['slot_hover']
324            elif slot_data is None:
325                color = self.colors['empty_slot']
326            else:
327                color = self.colors['slot']
328                
329            pygame.draw.rect(self.screen, color, rect)
330            pygame.draw.rect(self.screen, self.colors['text'], rect, 2)
331            
332            # Slot number
333            slot_text = self.font.render(f"Slot {slot_num}", True, self.colors['text'])
334            self.screen.blit(slot_text, (rect.x + 10, rect.y + 5))
335            
336            # Slot content
337            if slot_data:
338                # Save name
339                name_text = self.small_font.render(slot_data.save_name, True, self.colors['text'])
340                self.screen.blit(name_text, (rect.x + 120, rect.y + 5))
341                
342                # Details
343                details = f"Turn {slot_data.turn_number} | Player {slot_data.current_player} | "
344                details += f"{slot_data.knight_count} units | {slot_data.timestamp}"
345                details_text = self.small_font.render(details, True, self.colors['text_dim'])
346                self.screen.blit(details_text, (rect.x + 120, rect.y + 30))
347            else:
348                empty_text = self.small_font.render("Empty", True, self.colors['text_dim'])
349                self.screen.blit(empty_text, (rect.x + 120, rect.y + 20))
350                
351        # Remove clipping
352        self.screen.set_clip(None)
353        
354        # Draw scrollbar if needed
355        if self._is_scrollbar_visible():
356            # Scrollbar background
357            scrollbar_bg_rect = pygame.Rect(self.scrollbar_x, self.scrollbar_y, 
358                                           self.scrollbar_width, self.scrollbar_height)
359            pygame.draw.rect(self.screen, self.colors['scrollbar_bg'], scrollbar_bg_rect)
360            
361            # Scrollbar handle
362            handle_rect = self._get_scrollbar_rect()
363            pygame.draw.rect(self.screen, self.colors['scrollbar'], handle_rect)
364            pygame.draw.rect(self.screen, self.colors['text'], handle_rect, 1)
365                
366        # Draw save name input (save mode only)
367        if self.action == SaveLoadAction.SAVE:
368            # Input field
369            input_color = self.colors['input_active'] if self.input_active else self.colors['input_bg']
370            pygame.draw.rect(self.screen, input_color, self.input_rect)
371            pygame.draw.rect(self.screen, self.colors['text'], self.input_rect, 2)
372            
373            # Label
374            label_text = self.small_font.render("Save Name (optional):", True, self.colors['text'])
375            self.screen.blit(label_text, (self.input_rect.x, self.input_rect.y - 25))
376            
377            # Input text
378            if self.save_name_input:
379                input_surface = self.font.render(self.save_name_input, True, self.colors['text'])
380            else:
381                input_surface = self.font.render("Turn X", True, self.colors['text_dim'])
382            input_rect = input_surface.get_rect(midleft=(self.input_rect.x + 10, self.input_rect.centery))
383            self.screen.blit(input_surface, input_rect)
384            
385        # Draw confirmation dialog if needed
386        if self.confirm_overwrite or self.confirm_delete:
387            self._draw_confirmation_dialog()
388        else:
389            # Draw regular buttons
390            self._draw_buttons()
