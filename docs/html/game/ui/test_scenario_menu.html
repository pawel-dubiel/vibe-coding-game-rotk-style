 89    def handle_event(self, event: pygame.event.Event) -> Optional[ScenarioType]:
 90        """Handle input events"""
 91        if not self.visible:
 92            return None
 93            
 94        if event.type == pygame.KEYDOWN:
 95            if event.key == pygame.K_ESCAPE:
 96                self.hide()
 97                return None
 98            elif event.key == pygame.K_UP:
 99                self.hovered_index = max(0, self.hovered_index - 1)
100                self._ensure_visible()
101            elif event.key == pygame.K_DOWN:
102                self.hovered_index = min(len(self.scenario_list) - 1, self.hovered_index + 1)
103                self._ensure_visible()
104            elif event.key in (pygame.K_RETURN, pygame.K_SPACE):
105                if 0 <= self.hovered_index < len(self.scenario_list):
106                    self.selected_scenario = self.scenario_list[self.hovered_index][0]
107                    return self.selected_scenario
108                    
109        elif event.type == pygame.MOUSEMOTION:
110            mouse_pos = pygame.mouse.get_pos()
111            # Check scenario items
112            if self.clip_rect.collidepoint(mouse_pos):
113                relative_y = mouse_pos[1] - self.list_y + self.scroll_offset
114                index = relative_y // (self.item_height + self.item_spacing)
115                if 0 <= index < len(self.scenario_list):
116                    self.hovered_index = index
117                    
118        elif event.type == pygame.MOUSEBUTTONDOWN:
119            mouse_pos = pygame.mouse.get_pos()
120            if event.button == 1:  # Left click
121                # Check scenario items
122                if self.clip_rect.collidepoint(mouse_pos):
123                    relative_y = mouse_pos[1] - self.list_y + self.scroll_offset
124                    index = relative_y // (self.item_height + self.item_spacing)
125                    if 0 <= index < len(self.scenario_list):
126                        self.selected_scenario = self.scenario_list[index][0]
127                        return self.selected_scenario
128                        
129                # Check scrollbar
130                if self._is_scrollbar_visible():
131                    scrollbar_rect = self._get_scrollbar_rect()
132                    if scrollbar_rect.collidepoint(mouse_pos):
133                        self.dragging_scrollbar = True
134                        self.drag_offset = mouse_pos[1] - scrollbar_rect.y
135                        
136            elif event.button == 4:  # Mouse wheel up
137                self.scroll_offset = max(0, self.scroll_offset - (self.item_height + self.item_spacing))
138            elif event.button == 5:  # Mouse wheel down
139                max_scroll = self._get_max_scroll()
140                self.scroll_offset = min(max_scroll, self.scroll_offset + (self.item_height + self.item_spacing))
141                
142        elif event.type == pygame.MOUSEBUTTONUP:
143            if event.button == 1:
144                self.dragging_scrollbar = False
145                
146        elif event.type == pygame.MOUSEMOTION and hasattr(self, 'dragging_scrollbar') and self.dragging_scrollbar:
147            mouse_pos = pygame.mouse.get_pos()
148            # Calculate new scroll position based on mouse position
149            scrollbar_track_height = self.scrollbar_height - self._get_scrollbar_handle_height()
150            new_y = mouse_pos[1] - self.scrollbar_y - self.drag_offset
151            scroll_ratio = new_y / scrollbar_track_height if scrollbar_track_height > 0 else 0
152            scroll_ratio = max(0, min(1, scroll_ratio))
153            self.scroll_offset = int(scroll_ratio * self._get_max_scroll())
154                    
155        return None
156        
157    def _get_max_scroll(self):
158        """Get maximum scroll offset"""
159        total_height = len(self.scenario_list) * (self.item_height + self.item_spacing)
160        return max(0, total_height - self.visible_height)
161        
162    def _is_scrollbar_visible(self):
163        """Check if scrollbar should be visible"""
164        return len(self.scenario_list) > self.max_visible_items
165        
166    def _get_scrollbar_handle_height(self):
167        """Calculate scrollbar handle height"""
168        if not self._is_scrollbar_visible():
169            return 0
170        total_height = len(self.scenario_list) * (self.item_height + self.item_spacing)
171        ratio = self.visible_height / total_height if total_height > 0 else 1
172        return max(30, int(self.scrollbar_height * ratio))
173        
174    def _get_scrollbar_rect(self):
175        """Get scrollbar handle rectangle"""
176        handle_height = self._get_scrollbar_handle_height()
177        max_scroll = self._get_max_scroll()
178        
179        if max_scroll > 0:
180            scroll_ratio = self.scroll_offset / max_scroll
181        else:
182            scroll_ratio = 0
183            
184        handle_y = self.scrollbar_y + int((self.scrollbar_height - handle_height) * scroll_ratio)
185        return pygame.Rect(self.scrollbar_x, handle_y, self.scrollbar_width, handle_height)
186        
187    def _ensure_visible(self):
188        """Ensure the hovered item is visible"""
189        item_y = self.hovered_index * (self.item_height + self.item_spacing)
190        
191        # If item is above visible area
192        if item_y < self.scroll_offset:
193            self.scroll_offset = item_y
194            
195        # If item is below visible area
196        elif item_y + self.item_height > self.scroll_offset + self.visible_height:
197            self.scroll_offset = item_y + self.item_height - self.visible_height
198

200        """Draw the menu"""

246            self.screen.blit(name_text, name_rect)
247

249            desc_text = self.desc_font.render(scenario.description, True, self.colors['text_dim'])

251            self.screen.blit(desc_text, desc_rect)
252            
253        # Remove clipping
254        self.screen.set_clip(None)
255        
256        # Draw scrollbar if needed
257        if self._is_scrollbar_visible():
258            # Scrollbar background
259            scrollbar_bg_rect = pygame.Rect(self.scrollbar_x, self.scrollbar_y, 
260                                           self.scrollbar_width, self.scrollbar_height)
261            pygame.draw.rect(self.screen, self.colors['scrollbar_bg'], scrollbar_bg_rect)
262            
263            # Scrollbar handle
264            handle_rect = self._get_scrollbar_rect()
265            pygame.draw.rect(self.screen, self.colors['scrollbar'], handle_rect)
266            pygame.draw.rect(self.screen, self.colors['border'], handle_rect, 1)
267            
268        # Draw instructions
269        inst_text = self.desc_font.render("Use arrow keys or mouse to select, scroll wheel to scroll, Enter to confirm, ESC to cancel", 
270                                        True, self.colors['text_dim'])
271        inst_rect = inst_text.get_rect(centerx=self.panel_rect.centerx, 
272                                      y=self.panel_rect.bottom - 40)
273        self.screen.blit(inst_text, inst_rect)

 90    def handle_event(self, event: pygame.event.Event) -> Optional[ScenarioType]:
 91        """Handle input events"""
 92        if not self.visible:
 93            return None
 94            
 95        if event.type == pygame.KEYDOWN:
 96            if event.key == pygame.K_ESCAPE:
 97                self.hide()
 98                return None
 99            elif event.key == pygame.K_UP:
100                self.hovered_index = max(0, self.hovered_index - 1)
101                self._ensure_visible()
102            elif event.key == pygame.K_DOWN:
103                self.hovered_index = min(len(self.scenario_list) - 1, self.hovered_index + 1)
104                self._ensure_visible()
105            elif event.key in (pygame.K_RETURN, pygame.K_SPACE):
106                if 0 <= self.hovered_index < len(self.scenario_list):
107                    self.selected_scenario = self.scenario_list[self.hovered_index][0]
108                    return self.selected_scenario
109                    
110        elif event.type == pygame.MOUSEMOTION:
111            mouse_pos = pygame.mouse.get_pos()
112            # Check scenario items
113            if self.clip_rect.collidepoint(mouse_pos):
114                relative_y = mouse_pos[1] - self.list_y + self.scroll_offset
115                index = relative_y // (self.item_height + self.item_spacing)
116                if 0 <= index < len(self.scenario_list):
117                    self.hovered_index = index
118                    
119        elif event.type == pygame.MOUSEBUTTONDOWN:
120            mouse_pos = pygame.mouse.get_pos()
121            if event.button == 1:  # Left click
122                # Check scenario items
123                if self.clip_rect.collidepoint(mouse_pos):
124                    relative_y = mouse_pos[1] - self.list_y + self.scroll_offset
125                    index = relative_y // (self.item_height + self.item_spacing)
126                    if 0 <= index < len(self.scenario_list):
127                        self.selected_scenario = self.scenario_list[index][0]
128                        return self.selected_scenario
129                        
130                # Check scrollbar
131                if self._is_scrollbar_visible():
132                    scrollbar_rect = self._get_scrollbar_rect()
133                    if scrollbar_rect.collidepoint(mouse_pos):
134                        self.dragging_scrollbar = True
135                        self.drag_offset = mouse_pos[1] - scrollbar_rect.y
136                        
137            elif event.button == 4:  # Mouse wheel up
138                self.scroll_offset = max(0, self.scroll_offset - (self.item_height + self.item_spacing))
139            elif event.button == 5:  # Mouse wheel down
140                max_scroll = self._get_max_scroll()
141                self.scroll_offset = min(max_scroll, self.scroll_offset + (self.item_height + self.item_spacing))
142                
143        elif event.type == pygame.MOUSEBUTTONUP:
144            if event.button == 1:
145                self.dragging_scrollbar = False
146                
147        elif event.type == pygame.MOUSEMOTION and hasattr(self, 'dragging_scrollbar') and self.dragging_scrollbar:
148            mouse_pos = pygame.mouse.get_pos()
149            # Calculate new scroll position based on mouse position
150            scrollbar_track_height = self.scrollbar_height - self._get_scrollbar_handle_height()
151            new_y = mouse_pos[1] - self.scrollbar_y - self.drag_offset
152            scroll_ratio = new_y / scrollbar_track_height if scrollbar_track_height > 0 else 0
153            scroll_ratio = max(0, min(1, scroll_ratio))
154            self.scroll_offset = int(scroll_ratio * self._get_max_scroll())
155                    
156        return None
