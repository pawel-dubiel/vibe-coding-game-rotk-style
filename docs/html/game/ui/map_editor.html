120    def handle_event(self, event: pygame.event.Event) -> Optional[str]:
121        """Handle dialog events. Returns 'ok', 'cancel', or None"""
122        if event.type == pygame.MOUSEBUTTONDOWN and event.button == 1:
123            mouse_pos = event.pos
124            
125            # Check buttons
126            if self.ok_button.collidepoint(mouse_pos):
127                self._apply_changes()
128                return 'ok'
129            elif self.cancel_button.collidepoint(mouse_pos):
130                return 'cancel'
131            
132            # Check field clicks
133            self.active_field = None
134            for field_name, rect in self.field_rects.items():
135                if rect.collidepoint(mouse_pos):
136                    self.active_field = field_name
137                    if self.fields[field_name]['type'] == 'text':
138                        self.text_input = self.fields[field_name]['value']
139                    elif self.fields[field_name]['type'] == 'number':
140                        self.text_input = self.fields[field_name]['value']
141                    elif self.fields[field_name]['type'] == 'dropdown':
142                        self._cycle_dropdown_value(field_name)
143                    break
144        
145        elif event.type == pygame.KEYDOWN:
146            if event.key == pygame.K_ESCAPE:
147                return 'cancel'
148            elif event.key == pygame.K_RETURN:
149                if self.active_field:
150                    self._commit_field_value()
151                else:
152                    self._apply_changes()
153                    return 'ok'
154            elif event.key == pygame.K_TAB:
155                self._next_field()
156            elif self.active_field and self.fields[self.active_field]['type'] in ['text', 'number']:
157                self._handle_text_input(event)
158        
159        return None

161    def _cycle_dropdown_value(self, field_name: str):
162        """Cycle through dropdown options"""
163        field_data = self.fields[field_name]
164        current_value = field_data['value']
165        options = field_data['options']
166        
167        try:
168            current_index = options.index(current_value)
169            next_index = (current_index + 1) % len(options)
170            field_data['value'] = options[next_index]
171        except ValueError:
172            # Current value not in options, set to first option
173            field_data['value'] = options[0]

175    def _handle_text_input(self, event: pygame.event.Event):
176        """Handle text input for active field"""
177        if event.key == pygame.K_BACKSPACE:
178            self.text_input = self.text_input[:-1]
179        elif event.unicode.isprintable():
180            # For number fields, only allow digits and one decimal point
181            if self.fields[self.active_field]['type'] == 'number':
182                if event.unicode.isdigit() or (event.unicode == '.' and '.' not in self.text_input):
183                    self.text_input += event.unicode
184            else:
185                self.text_input += event.unicode

187    def _commit_field_value(self):
188        """Commit the current text input to the active field"""
189        if self.active_field and self.fields[self.active_field]['type'] in ['text', 'number']:
190            self.fields[self.active_field]['value'] = self.text_input
191            self.active_field = None
192            self.text_input = ""

194    def _next_field(self):
195        """Move to next editable field"""
196        field_names = list(self.fields.keys())
197        if self.active_field:
198            try:
199                current_index = field_names.index(self.active_field)
200                next_index = (current_index + 1) % len(field_names)
201                self.active_field = field_names[next_index]
202            except ValueError:
203                self.active_field = field_names[0]
204        else:
205            self.active_field = field_names[0]
206        
207        # Set text input for text/number fields
208        if self.fields[self.active_field]['type'] in ['text', 'number']:
209            self.text_input = self.fields[self.active_field]['value']

417    def handle_event(self, event: pygame.event.Event) -> Optional[Dict[str, Any]]:
418        """Handle editor events"""
419        if not self.visible:
420            return None
421        
422        # Handle city edit dialog if open
423        if self.city_edit_dialog:
424            result = self.city_edit_dialog.handle_event(event)
425            if result == 'ok':
426                self._save_state()  # Save after editing
427                self.city_edit_dialog = None
428                return None
429            elif result == 'cancel':
430                self.city_edit_dialog = None
431                return None
432            else:
433                return None  # Dialog is handling the event
434        
435        if event.type == pygame.KEYDOWN:
436            if event.key == pygame.K_ESCAPE:
437                return {'action': 'back'}
438            elif event.key == pygame.K_s and pygame.key.get_pressed()[pygame.K_LCTRL]:
439                self._save_map()
440                return None
441            elif event.key == pygame.K_z and pygame.key.get_pressed()[pygame.K_LCTRL]:
442                if pygame.key.get_pressed()[pygame.K_LSHIFT]:
443                    self._redo()
444                else:
445                    self._undo()
446                return None
447        
448        elif event.type == pygame.MOUSEBUTTONDOWN:
449            if event.button == 1:  # Left click
450                self.mouse_pressed = True
451                self._handle_left_click(event.pos)
452            elif event.button == 3:  # Right click
453                self._handle_right_click(event.pos)
454        
455        elif event.type == pygame.MOUSEBUTTONUP:
456            if event.button == 1:
457                self.mouse_pressed = False
458                self.dragging = False
459                self.drag_start = None
460        
461        elif event.type == pygame.MOUSEMOTION:
462            self.mouse_pos = event.pos
463            if self.mouse_pressed and self.current_tool == EditorTool.TERRAIN_PAINT:
464                self._handle_terrain_paint(event.pos)
465            elif self.dragging and self.current_tool == EditorTool.PAN:
466                self._handle_pan_drag(event.pos)
467        
468        return None
