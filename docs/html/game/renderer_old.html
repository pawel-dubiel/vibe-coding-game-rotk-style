 54    def _draw_dotted_line(self, surface, start_pos, end_pos, color, width=1, dash_length=5):
 55        """Draw a dotted line between two points"""
 56        x1, y1 = start_pos
 57        x2, y2 = end_pos
 58        
 59        # Calculate distance and direction
 60        dx = x2 - x1
 61        dy = y2 - y1
 62        distance = math.sqrt(dx**2 + dy**2)
 63        
 64        if distance == 0:
 65            return
 66            
 67        # Normalize direction
 68        dx /= distance
 69        dy /= distance
 70        
 71        # Draw dashes
 72        drawn = 0
 73        draw_dash = True
 74        
 75        while drawn < distance:
 76            if draw_dash:
 77                # Calculate dash end point
 78                dash_end = min(drawn + dash_length, distance)
 79                
 80                # Draw the dash
 81                start_x = x1 + dx * drawn
 82                start_y = y1 + dy * drawn
 83                end_x = x1 + dx * dash_end
 84                end_y = y1 + dy * dash_end
 85                
 86                pygame.draw.line(surface, color, 
 87                               (int(start_x), int(start_y)), 
 88                               (int(end_x), int(end_y)), 
 89                               width)
 90            
 91            drawn += dash_length
 92            draw_dash = not draw_dash

 55    def _draw_dotted_line(self, surface, start_pos, end_pos, color, width=1, dash_length=5):
 56        """Draw a dotted line between two points"""
 57        x1, y1 = start_pos
 58        x2, y2 = end_pos
 59        
 60        # Calculate distance and direction
 61        dx = x2 - x1
 62        dy = y2 - y1
 63        distance = math.sqrt(dx**2 + dy**2)
 64        
 65        if distance == 0:
 66            return
 67            
 68        # Normalize direction
 69        dx /= distance
 70        dy /= distance
 71        
 72        # Draw dashes
 73        drawn = 0
 74        draw_dash = True
 75        
 76        while drawn < distance:
 77            if draw_dash:
 78                # Calculate dash end point
 79                dash_end = min(drawn + dash_length, distance)
 80                
 81                # Draw the dash
 82                start_x = x1 + dx * drawn
 83                start_y = y1 + dy * drawn
 84                end_x = x1 + dx * dash_end
 85                end_y = y1 + dy * dash_end
 86                
 87                pygame.draw.line(surface, color, 
 88                               (int(start_x), int(start_y)), 
 89                               (int(end_x), int(end_y)), 
 90                               width)
 91            
 92            drawn += dash_length
 93            draw_dash = not draw_dash
