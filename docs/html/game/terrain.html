216    def __init__(self, width: int, height: int, seed: Optional[int] = None):
217        self.width = width
218        self.height = height
219        self.seed = seed or random.randint(0, 1000000)
220        random.seed(self.seed)
221        self._noise_cache = {}

223    def _simple_noise(self, x: float, y: float, seed_offset: int = 0) -> float:
224        """Simple pseudo-random noise function"""
225        # Create a hash from coordinates
226        hash_val = hash((int(x * 1000), int(y * 1000), self.seed + seed_offset))
227        # Convert to float in range -1 to 1
228        return (hash_val % 10000) / 5000.0 - 1.0
229        
230    @staticmethod
231    def _interpolate(a: float, b: float, t: float) -> float:
232        """Smooth interpolation between two values"""
233        # Smoothstep interpolation
234        t = t * t * (3 - 2 * t)
235        return a + t * (b - a)
236        
237    def _perlin_noise(self, x: float, y: float, seed_offset: int = 0) -> float:
238        """Simple Perlin-like noise implementation"""
239        # Grid coordinates
240        x0 = int(math.floor(x))
241        y0 = int(math.floor(y))
242        x1 = x0 + 1
243        y1 = y0 + 1
244        
245        # Interpolation weights
246        sx = x - x0
247        sy = y - y0
248        
249        # Get noise values at grid corners
250        n00 = self._simple_noise(x0, y0, seed_offset)
251        n10 = self._simple_noise(x1, y0, seed_offset)
252        n01 = self._simple_noise(x0, y1, seed_offset)
253        n11 = self._simple_noise(x1, y1, seed_offset)
254        
255        # Interpolate
256        nx0 = self._interpolate(n00, n10, sx)
257        nx1 = self._interpolate(n01, n11, sx)
258        
259        return self._interpolate(nx0, nx1, sy)

261    def generate_height_map(self, scale: float = 0.1, octaves: int = 4) -> List[List[float]]:
262        """Generate height map using Perlin noise"""
263        height_map = []
264        
265        for y in range(self.height):
266            row = []
267            for x in range(self.width):
268                # Use multiple octaves for more realistic terrain
269                value = 0
270                amplitude = 1
271                frequency = scale
272                
273                for octave in range(octaves):
274                    value += self._perlin_noise(
275                        x * frequency,
276                        y * frequency,
277                        octave * 1000
278                    ) * amplitude
279                    amplitude *= 0.5
280                    frequency *= 2
281                    
282                # Normalize to 0-1 range
283                row.append((value + 1) / 2)
284            height_map.append(row)
285            
286        return height_map

288    def generate_moisture_map(self, scale: float = 0.15) -> List[List[float]]:
289        """Generate moisture map for vegetation distribution"""

353    def generate_rivers(self, terrain_grid: List[List[Terrain]], 
354                       height_map: List[List[float]]) -> None:
355        """Generate rivers that flow from high to low elevation"""
356        # Find potential stream sources (moderate elevation)
357        sources = []
358        for y in range(self.height):
359            for x in range(self.width):
360                # Look for moderate elevation areas for stream sources (less extreme than rivers)
361                if 0.5 < height_map[y][x] < 0.8 and random.random() < 0.03:  # Increased chance for more streams
362                    sources.append((x, y))
363                    
364        # Flow rivers downhill
365        for source_x, source_y in sources:
366            x, y = source_x, source_y
367            river_tiles = set()
368            
369            while 0 <= x < self.width and 0 <= y < self.height:
370                river_tiles.add((x, y))
371                
372                # Find lowest neighbor
373                neighbors = [
374                    (x-1, y), (x+1, y), (x, y-1), (x, y+1),
375                    (x-1, y-1), (x+1, y-1), (x-1, y+1), (x+1, y+1)
376                ]
377                
378                valid_neighbors = [
379                    (nx, ny) for nx, ny in neighbors
380                    if 0 <= nx < self.width and 0 <= ny < self.height
381                    and (nx, ny) not in river_tiles
382                ]
383                
384                if not valid_neighbors:
385                    break
386                    
387                # Flow to lowest neighbor
388                lowest = min(valid_neighbors, key=lambda pos: height_map[pos[1]][pos[0]])
389                
390                # Stop if we hit water or go uphill
391                if height_map[lowest[1]][lowest[0]] >= height_map[y][x]:
392                    break
393                if terrain_grid[lowest[1]][lowest[0]].type in [TerrainType.WATER, TerrainType.DEEP_WATER]:
394                    break
395                    
396                x, y = lowest
397                
398                # Add stream feature for battle-appropriate water (avoid blocking rivers)
399                if terrain_grid[y][x].can_support_feature(TerrainFeature.STREAM):
400                    terrain_grid[y][x].feature = TerrainFeature.STREAM

402    def generate_roads(self, terrain_grid: List[List[Terrain]], 
403                      important_points: List[Tuple[int, int]]) -> None:
404        """Generate roads connecting important points (like castles)"""
405        if len(important_points) < 2:
406            return
407            
408        # Connect each point to the nearest other point
409        for i, start in enumerate(important_points):
410            # Find nearest unconnected point
411            best_end = None
412            best_distance = float('inf')
413            
414            for j, end in enumerate(important_points):
415                if i != j:
416                    dist = abs(end[0] - start[0]) + abs(end[1] - start[1])
417                    if dist < best_distance:
418                        best_distance = dist
419                        best_end = end
420                        
421            if best_end:
422                self._create_road(terrain_grid, start, best_end)
423                
424    def _create_road(self, terrain_grid: List[List[Terrain]], 
425                    start: Tuple[int, int], end: Tuple[int, int]) -> None:
426        """Create a road between two points using A* pathfinding"""
427        from heapq import heappush, heappop
428        
429        # Simple A* to find good path for road
430        open_set = [(0, start)]
431        came_from = {}
432        g_score = {start: 0}
433        
434        while open_set:
435            _, current = heappop(open_set)
436            
437            if current == end:
438                # Reconstruct path
439                path = []
440                while current in came_from:
441                    path.append(current)
442                    current = came_from[current]
443                path.append(start)
444                
445                # Add road to tiles
446                for x, y in path:
447                    tile = terrain_grid[y][x]
448                    if tile.can_support_feature(TerrainFeature.ROAD):
449                        tile.feature = TerrainFeature.ROAD
450                    elif tile.type == TerrainType.WATER and tile.can_support_feature(TerrainFeature.BRIDGE):
451                        tile.feature = TerrainFeature.BRIDGE
452                return
453                
454            # Check neighbors
455            x, y = current
456            for dx, dy in [(0, 1), (0, -1), (1, 0), (-1, 0)]:
457                nx, ny = x + dx, y + dy
458                if 0 <= nx < self.width and 0 <= ny < self.height:
459                    neighbor = (nx, ny)
460                    
461                    # Cost based on terrain
462                    terrain = terrain_grid[ny][nx]
463                    if not terrain.passable and terrain.type != TerrainType.WATER:
464                        continue
465                        
466                    move_cost = 1
467                    if terrain.type in [TerrainType.FOREST, TerrainType.DENSE_FOREST]:
468                        move_cost = 3  # Prefer to avoid forests
469                    elif terrain.type == TerrainType.WATER:
470                        move_cost = 5  # Expensive to bridge
471                        
472                    tentative_g = g_score[current] + move_cost
473                    
474                    if neighbor not in g_score or tentative_g < g_score[neighbor]:
475                        came_from[neighbor] = current
476                        g_score[neighbor] = tentative_g
477                        f_score = tentative_g + abs(end[0] - nx) + abs(end[1] - ny)
478                        heappush(open_set, (f_score, neighbor))

511                    if distance <= radius:

513                        if 0 <= x < self.width and 0 <= y < self.height:
514                            current_terrain = terrain_grid[y][x]
515                            
516                            # Only convert plains or existing light forest
517                            if current_terrain.type in [TerrainType.PLAINS, TerrainType.LIGHT_FOREST]:
518                                # Determine forest type based on distance from center
519                                if distance == 0:
520                                    # Center: dense forest
521                                    terrain_grid[y][x] = Terrain(TerrainType.DENSE_FOREST)
522                                elif distance <= 1:
523                                    # Inner ring: regular forest
524                                    terrain_grid[y][x] = Terrain(TerrainType.FOREST)

735                           end_x: int, end_y: int) -> None:

737        # Simple line drawing algorithm
738        dx = abs(end_x - start_x)
739        dy = abs(end_y - start_y)
740        sx = 1 if start_x < end_x else -1
741        sy = 1 if start_y < end_y else -1
742        err = dx - dy
743        
744        x, y = start_x, start_y
745        
746        while True:
747            # Add stream with some randomness
748            if 0 <= x < self.width and 0 <= y < self.height:
749                if terrain_grid[y][x].type == TerrainType.PLAINS:
750                    terrain_grid[y][x] = Terrain(TerrainType.PLAINS, TerrainFeature.STREAM)
751                
752                # Occasionally add adjacent stream tiles for wider streams
753                if random.random() < 0.3:
754                    for dx_adj, dy_adj in [(0, 1), (0, -1), (1, 0), (-1, 0)]:
755                        adj_x, adj_y = x + dx_adj, y + dy_adj
756                        if (0 <= adj_x < self.width and 0 <= adj_y < self.height and
757                            terrain_grid[adj_y][adj_x].type == TerrainType.PLAINS and
758                            random.random() < 0.5):
759                            terrain_grid[adj_y][adj_x] = Terrain(TerrainType.PLAINS, TerrainFeature.STREAM)
760            
761            if x == end_x and y == end_y:
762                break
763                
764            e2 = 2 * err
765            if e2 > -dy:
766                err -= dy
767                x += sx
768            if e2 < dx:
769                err += dx
770                y += sy

438            if current == end:

452                        tile.feature = TerrainFeature.BRIDGE

455            # Check neighbors
456            x, y = current
