 70    def load_scenario(cls, filename: str) -> ScenarioDefinition:
 71        """Load a scenario from a JSON file"""
 72        # Check if it's a full path or just a filename
 73        if os.path.exists(filename):
 74            filepath = filename
 75        else:
 76            # Look in the test_scenarios directory
 77            base_dir = os.path.dirname(os.path.dirname(os.path.abspath(__file__)))
 78            filepath = os.path.join(base_dir, 'test_scenarios', filename)
 79            
 80            # Add .json extension if not present
 81            if not filepath.endswith('.json'):
 82                filepath += '.json'
 83        
 84        with open(filepath, 'r') as f:
 85            data = json.load(f)
 86        
 87        return ScenarioDefinition(
 88            name=data['name'],
 89            description=data['description'],
 90            board_size=tuple(data['board_size']),
 91            terrain_base=data.get('terrain', {}).get('base', 'plains'),
 92            terrain_tiles=data.get('terrain', {}).get('tiles', []),
 93            units=data.get('units', []),
 94            castles=data.get('castles', []),
 95            victory_conditions=data.get('victory_conditions', {})
 96        )

100        """List all available scenario files"""
101        base_dir = os.path.dirname(os.path.dirname(os.path.abspath(__file__)))

114    @classmethod
115    def apply_to_game_state(cls, scenario: ScenarioDefinition, game_state):
116        """Apply a scenario definition to a game state"""
117        # Clear existing units and castles
118        game_state.knights.clear()
119        game_state.castles.clear()
120        
121        # Set up terrain
122        if hasattr(game_state, 'terrain_map') and game_state.terrain_map:
123            # Set base terrain
124            base_terrain = cls.TERRAIN_MAPPING.get(scenario.terrain_base, TerrainType.PLAINS)
125            for x in range(game_state.board_width):
126                for y in range(game_state.board_height):
127                    game_state.terrain_map.set_terrain(x, y, base_terrain)
128            
129            # Apply specific terrain tiles
130            for tile in scenario.terrain_tiles:
131                terrain_type = cls.TERRAIN_MAPPING.get(tile['type'], TerrainType.PLAINS)
132                game_state.terrain_map.set_terrain(tile['x'], tile['y'], terrain_type)
133        
134        # Create units
135        for unit_def in scenario.units:
136            unit_type = cls.UNIT_TYPE_MAPPING.get(unit_def['type'], KnightClass.WARRIOR)
137            unit = UnitFactory.create_unit(
138                name=unit_def['name'],
139                unit_class=unit_type,
140                x=unit_def['x'],
141                y=unit_def['y']
142            )
143            unit.player_id = unit_def['player']
144            
145            # Set optional attributes
146            if 'facing' in unit_def:
147                facing_dir = cls.FACING_MAPPING.get(unit_def['facing'], FacingDirection.EAST)
148                unit.facing.facing = facing_dir
149            
150            if 'health' in unit_def:
151                health_percent = unit_def['health'] / 100.0
152                unit.stats.stats.current_soldiers = int(unit.max_soldiers * health_percent)
153            
154            if 'morale' in unit_def:
155                unit.morale = unit_def['morale']
156            
157            game_state.knights.append(unit)
158        
159        # Create castles
160        for castle_def in scenario.castles:
161            castle = Castle(castle_def['x'], castle_def['y'], castle_def['player'])
162            game_state.castles.append(castle)
163        
164        # Update fog of war if present
165        if hasattr(game_state, '_update_all_fog_of_war'):
166            game_state._update_all_fog_of_war()
167    
168    @classmethod
169    def create_test_function(cls, scenario_file: str):
170        """Create a test scenario setup function for the game"""
171        def setup_scenario(game_state):
172            scenario = cls.load_scenario(scenario_file)
173            cls.apply_to_game_state(scenario, game_state)
174            return scenario.name, scenario.description
175        
176        return setup_scenario
