 182        for behavior in self.behaviors.values():
 183            # Check exact match
 184            if type(behavior).__name__ == behavior_type:
 185                return behavior
 186            # Check if it's a subclass of the requested type
 187            behavior_class_name = type(behavior).__name__
 188            if behavior_type in behavior_class_name or behavior_class_name.endswith(behavior_type):
 189                return behavior
 190            # Check inheritance hierarchy
 191            for base_class in type(behavior).__mro__:
 192                if base_class.__name__ == behavior_type:
 193                    return behavior

 220    def check_routing(self, game_state, shock_bonus: float) -> bool:
 221        """Check if unit should start routing based on morale, cohesion, and conditions"""
 222        if game_state is None:
 223            raise ValueError("game_state is required for routing checks")
 224        if shock_bonus is None:
 225            raise ValueError("shock_bonus is required for routing checks")
 226        # Already routing
 227        if self.is_routing:
 228            return True
 229            
 230        from game.combat_config import CombatConfig
 231
 232        # Units with very low morale or cohesion start routing immediately
 233        if (self.morale <= CombatConfig.ROUTING_HARD_MORALE_THRESHOLD or
 234                self.cohesion <= CombatConfig.ROUTING_HARD_COHESION_THRESHOLD):
 235            self._start_routing(game_state)
 236            return True
 237            
 238        pressure_bonus = 0.0
 239        adjacent_enemies = 0
 240        adjacent_friendlies = 0
 241        for other in game_state.knights:
 242            if other == self or other.is_garrisoned:
 243                continue
 244            dx = abs(self.x - other.x)
 245            dy = abs(self.y - other.y)
 246            if dx <= 1 and dy <= 1 and (dx + dy > 0):
 247                if other.player_id != self.player_id:
 248                    adjacent_enemies += 1
 249                else:
 250                    adjacent_friendlies += 1
 251
 252        if adjacent_enemies > adjacent_friendlies:
 253            pressure_bonus = CombatConfig.ROUTING_PRESSURE_BONUS
 254
 255        route_chance = CombatConfig.calculate_routing_chance(
 256            morale=self.morale,
 257            cohesion=self.cohesion,
 258            pressure_bonus=pressure_bonus,
 259            shock_bonus=shock_bonus
 260        )
 261        if route_chance > 0:
 262            import random
 263            if random.random() < (route_chance / 100.0):
 264                self._start_routing(game_state)
 265                return True
 266                
 267        return False

 269    def _start_routing(self, game_state=None):
 270        """Start routing and attempt immediate flee movement"""
 271        was_routing = self.is_routing
 272        self.is_routing = True

 276        
 277        # If game state is available and unit just started routing, try to auto-flee
 278        if not was_routing and game_state is not None:
 279            self._attempt_auto_routing_movement(game_state)

 281    def _attempt_auto_routing_movement(self, game_state):
 282        """Attempt automatic routing movement away from enemies"""
 283        if not self.is_routing:
 284            return False
 285            
 286        # Check if unit has movement behavior and can move
 287        movement_behavior = self.get_behavior('MovementBehavior')
 288        if not movement_behavior:
 289            return False
 290            
 291        # Get routing movement options
 292        routing_moves = movement_behavior._get_routing_moves(self, game_state)
 293        if not routing_moves:
 294            return False
 295            
 296        # Choose the best routing move (furthest from nearest enemy)
 297        enemies = [k for k in game_state.knights if k.player_id != self.player_id]
 298        if not enemies:
 299            return False
 300            
 301        nearest_enemy = min(enemies, key=lambda e: abs(e.x - self.x) + abs(e.y - self.y))
 302        
 303        # Find the move that gets furthest from nearest enemy
 304        best_move = None
 305        best_distance = 0
 306        
 307        for move_x, move_y in routing_moves:
 308            distance = abs(move_x - nearest_enemy.x) + abs(move_y - nearest_enemy.y)
 309            if distance > best_distance:
 310                best_distance = distance
 311                best_move = (move_x, move_y)
 312        
 313        # Execute the routing movement if we found a good move
 314        if best_move:
 315            old_x, old_y = self.x, self.y
 316            new_x, new_y = best_move
 317            
 318            # Check if position is valid and not occupied
 319            if (0 <= new_x < game_state.board_width and 
 320                0 <= new_y < game_state.board_height and
 321                not game_state.get_knight_at(new_x, new_y)):
 322                
 323                # Execute the routing movement directly
 324                self.x = new_x
 325                self.y = new_y
 326                
 327                # Consume some AP for the panic movement (but don't block future movement)
 328                if not getattr(game_state, 'disable_auto_routing_ap_cost', False):
 329                    self.action_points = max(0, self.action_points - 1)
 330                
 331                # Add message about routing movement
 332                if hasattr(game_state, 'add_message'):
 333                    game_state.add_message(f"{self.name} routs and flees!", priority=2)
 334                
 335                # Update facing to show the unit is fleeing away from enemy
 336                if hasattr(self, 'facing') and self.facing:
 337                    self.facing.update_facing_from_movement(old_x, old_y, new_x, new_y)
 338                
 339                return True
 340        
 341        return False

 343    def end_turn(self):
 344        """Reset for new turn"""
 345        self.action_points = self.max_action_points
 346        self.has_moved = False
 347        self.has_acted = False
 348        self.has_used_special = False
 349        
 350        # Regenerate will
 351        self.stats.regenerate_will(20)
 352        
 353        from game.combat_config import CombatConfig
 354
 355        # Apply base morale/cohesion regeneration
 356        if self.is_routing:
 357            base_morale_regen = CombatConfig.MORALE_REGEN_ROUTING
 358            base_cohesion_regen = CombatConfig.COHESION_REGEN_ROUTING
 359        else:
 360            base_morale_regen = CombatConfig.MORALE_REGEN_BASE
 361            base_cohesion_regen = CombatConfig.COHESION_REGEN_BASE
 362
 363        if self.is_engaged_in_combat:
 364            base_morale_regen *= CombatConfig.REGEN_ENGAGED_MULTIPLIER
 365            base_cohesion_regen *= CombatConfig.REGEN_ENGAGED_MULTIPLIER
 366
 367        # Apply general morale regeneration bonuses
 368        bonuses = self.generals.get_all_passive_bonuses(self)
 369        general_morale_regen = bonuses.get('morale_regen', 0)
 370        
 371        total_morale_regen = base_morale_regen + general_morale_regen
 372        self.stats.stats.morale = min(self.stats.stats.max_morale, self.stats.stats.morale + total_morale_regen)
 373        self.stats.stats.current_cohesion = min(
 374            self.stats.stats.max_cohesion,
 375            self.stats.stats.current_cohesion + base_cohesion_regen
 376        )
 377        
 378        # Check if routing unit recovers enough morale to stop routing
 379        self.has_rallied_this_turn = False
 380        
 381        # Check if unit has rallies remaining (Militia=0, Regular=1, Elite=3)
 382        can_rally = self.times_routed <= self.quality.max_rallies
 383        
 384        if (self.is_routing and can_rally and
 385                self.stats.stats.morale >= CombatConfig.RALLY_MORALE_THRESHOLD and
 386                self.stats.stats.current_cohesion >= CombatConfig.RALLY_COHESION_THRESHOLD):
 387            # Additional rally conditions for realism
 388            rally_chance = CombatConfig.RALLY_BASE_CHANCE
 389            
 390            # Better chance to rally with higher morale
 391            if self.stats.stats.morale >= CombatConfig.RALLY_HIGH_MORALE_THRESHOLD:
 392                rally_chance = CombatConfig.RALLY_HIGH_MORALE_CHANCE
 393
 394            if self.stats.stats.current_cohesion >= CombatConfig.RALLY_HIGH_COHESION_THRESHOLD:
 395                rally_chance *= CombatConfig.RALLY_HIGH_COHESION_MULTIPLIER
 396                
 397            # Reduced chance if heavily damaged
 398            soldier_ratio = self.soldiers / self.max_soldiers
 399            if soldier_ratio < 0.5:  # Less than 50% soldiers remaining
 400                rally_chance *= CombatConfig.RALLY_LOW_STRENGTH_MULTIPLIER
 401                
 402            import random
 403            if random.random() < rally_chance:
 404                self.is_routing = False
 405                self.has_rallied_this_turn = True
 406            
 407        # Reset temporary combat modifiers
 408        self.temp_damage_multiplier = 1.0
 409        self.temp_vulnerability = 1.0
 410        
 411        # Reset attack counter for new turn
 412        self.attacks_this_turn = 0
 413        
 414        # Update generals
 415        self.generals.on_turn_end()

 779    def can_charge(self, target, game_state):
 780        """Check if cavalry can charge the target"""
 781        if self.knight_class != KnightClass.CAVALRY:
 782            return False, "Only cavalry can charge"
 783        
 784        # Use will property which goes through stats component
 785        if self.will < 40:
 786            return False, "Not enough will (need 40)"
 787        
 788        if self.has_used_special:
 789            return False, "Already used special ability"
 790        
 791        if self.is_routing:
 792            return False, "Routing units cannot charge"
 793        
 794        # Must be adjacent (including diagonals)
 795        dx = abs(self.x - target.x)
 796        dy = abs(self.y - target.y)
 797        if not (dx <= 1 and dy <= 1 and (dx + dy > 0)):
 798            return False, "Must be adjacent to charge"
 799        
 800        # Check terrain restrictions for cavalry charges
 801        if game_state.terrain_map:
 802            # Cavalry cannot charge when on hills
 803            cavalry_terrain = game_state.terrain_map.get_terrain(self.x, self.y)
 804            if cavalry_terrain and cavalry_terrain.type.value.lower() == 'hills':
 805                return False, "Cannot charge from hills"
 806            
 807            # Cavalry cannot charge enemies on hills
 808            target_terrain = game_state.terrain_map.get_terrain(target.x, target.y)
 809            if target_terrain and target_terrain.type.value.lower() == 'hills':
 810                return False, "Cannot charge enemies on hills"
 811

 813        # The outcome (damage/push) depends on what's behind the target
 814        return True, "Can charge"

 181    def get_behavior(self, behavior_type: str) -> Optional[Behavior]:
 182        """Get a behavior by its type name (e.g. 'VisionBehavior')"""
 183        for behavior in self.behaviors.values():
 184            # Check exact match
 185            if type(behavior).__name__ == behavior_type:
 186                return behavior
 187            # Check if it's a subclass of the requested type
 188            behavior_class_name = type(behavior).__name__
 189            if behavior_type in behavior_class_name or behavior_class_name.endswith(behavior_type):
 190                return behavior
 191            # Check inheritance hierarchy
 192            for base_class in type(behavior).__mro__:
 193                if base_class.__name__ == behavior_type:
 194                    return behavior
 195        return None

 205    def take_casualties(self, amount: int, game_state) -> bool:
 206        """Apply casualties to unit"""
 207        if game_state is None:
 208            raise ValueError("game_state is required when applying casualties")
 209        # Routing units are harder to hit effectively (they're fleeing/scattered)
 210        if self.is_routing:
 211            amount = int(amount * 0.7)  # 30% damage reduction for routing units
 212            
 213        was_destroyed = self.stats.take_casualties(amount)
 214        
 215        # Check for automatic routing after casualties (only if not already routing)
 216        if not was_destroyed and not self.is_routing:
 217            self.check_routing(game_state, shock_bonus=0.0)
 218            
 219        return was_destroyed

 221    def check_routing(self, game_state, shock_bonus: float) -> bool:
 222        """Check if unit should start routing based on morale, cohesion, and conditions"""
 223        if game_state is None:
 224            raise ValueError("game_state is required for routing checks")
 225        if shock_bonus is None:
 226            raise ValueError("shock_bonus is required for routing checks")
 227        # Already routing
 228        if self.is_routing:
 229            return True
 230            
 231        from game.combat_config import CombatConfig
 232
 233        # Units with very low morale or cohesion start routing immediately
 234        if (self.morale <= CombatConfig.ROUTING_HARD_MORALE_THRESHOLD or
 235                self.cohesion <= CombatConfig.ROUTING_HARD_COHESION_THRESHOLD):
 236            self._start_routing(game_state)
 237            return True
 238            
 239        pressure_bonus = 0.0
 240        adjacent_enemies = 0
 241        adjacent_friendlies = 0
 242        for other in game_state.knights:
 243            if other == self or other.is_garrisoned:
 244                continue
 245            dx = abs(self.x - other.x)
 246            dy = abs(self.y - other.y)
 247            if dx <= 1 and dy <= 1 and (dx + dy > 0):
 248                if other.player_id != self.player_id:
 249                    adjacent_enemies += 1
 250                else:
 251                    adjacent_friendlies += 1
 252
 253        if adjacent_enemies > adjacent_friendlies:
 254            pressure_bonus = CombatConfig.ROUTING_PRESSURE_BONUS
 255
 256        route_chance = CombatConfig.calculate_routing_chance(
 257            morale=self.morale,
 258            cohesion=self.cohesion,
 259            pressure_bonus=pressure_bonus,
 260            shock_bonus=shock_bonus
 261        )
 262        if route_chance > 0:
 263            import random
 264            if random.random() < (route_chance / 100.0):
 265                self._start_routing(game_state)
 266                return True
 267                
 268        return False

 270    def _start_routing(self, game_state=None):
 271        """Start routing and attempt immediate flee movement"""
 272        was_routing = self.is_routing
 273        self.is_routing = True
 274        
 275        if not was_routing:
 276            self.times_routed += 1
 277        
 278        # If game state is available and unit just started routing, try to auto-flee
 279        if not was_routing and game_state is not None:
 280            self._attempt_auto_routing_movement(game_state)
 281    
 282    def _attempt_auto_routing_movement(self, game_state):
 283        """Attempt automatic routing movement away from enemies"""
 284        if not self.is_routing:
 285            return False
 286            
 287        # Check if unit has movement behavior and can move
 288        movement_behavior = self.get_behavior('MovementBehavior')
 289        if not movement_behavior:
 290            return False
 291            
 292        # Get routing movement options
 293        routing_moves = movement_behavior._get_routing_moves(self, game_state)
 294        if not routing_moves:
 295            return False
 296            
 297        # Choose the best routing move (furthest from nearest enemy)
 298        enemies = [k for k in game_state.knights if k.player_id != self.player_id]
 299        if not enemies:
 300            return False
 301            
 302        nearest_enemy = min(enemies, key=lambda e: abs(e.x - self.x) + abs(e.y - self.y))
 303        
 304        # Find the move that gets furthest from nearest enemy
 305        best_move = None
 306        best_distance = 0
 307        
 308        for move_x, move_y in routing_moves:
 309            distance = abs(move_x - nearest_enemy.x) + abs(move_y - nearest_enemy.y)
 310            if distance > best_distance:
 311                best_distance = distance
 312                best_move = (move_x, move_y)
 313        
 314        # Execute the routing movement if we found a good move
 315        if best_move:
 316            old_x, old_y = self.x, self.y
 317            new_x, new_y = best_move
 318            
 319            # Check if position is valid and not occupied
 320            if (0 <= new_x < game_state.board_width and 
 321                0 <= new_y < game_state.board_height and
 322                not game_state.get_knight_at(new_x, new_y)):
 323                
 324                # Execute the routing movement directly
 325                self.x = new_x
 326                self.y = new_y
 327                
 328                # Consume some AP for the panic movement (but don't block future movement)
 329                if not getattr(game_state, 'disable_auto_routing_ap_cost', False):
 330                    self.action_points = max(0, self.action_points - 1)
 331                
 332                # Add message about routing movement
 333                if hasattr(game_state, 'add_message'):
 334                    game_state.add_message(f"{self.name} routs and flees!", priority=2)
 335                
 336                # Update facing to show the unit is fleeing away from enemy
 337                if hasattr(self, 'facing') and self.facing:
 338                    self.facing.update_facing_from_movement(old_x, old_y, new_x, new_y)
 339                
 340                return True
 341        
 342        return False

 344    def end_turn(self):
 345        """Reset for new turn"""
 346        self.action_points = self.max_action_points
 347        self.has_moved = False
 348        self.has_acted = False
 349        self.has_used_special = False
 350        
 351        # Regenerate will
 352        self.stats.regenerate_will(20)
 353        
 354        from game.combat_config import CombatConfig
 355
 356        # Apply base morale/cohesion regeneration
 357        if self.is_routing:
 358            base_morale_regen = CombatConfig.MORALE_REGEN_ROUTING
 359            base_cohesion_regen = CombatConfig.COHESION_REGEN_ROUTING
 360        else:
 361            base_morale_regen = CombatConfig.MORALE_REGEN_BASE
 362            base_cohesion_regen = CombatConfig.COHESION_REGEN_BASE
 363
 364        if self.is_engaged_in_combat:
 365            base_morale_regen *= CombatConfig.REGEN_ENGAGED_MULTIPLIER
 366            base_cohesion_regen *= CombatConfig.REGEN_ENGAGED_MULTIPLIER
 367
 368        # Apply general morale regeneration bonuses
 369        bonuses = self.generals.get_all_passive_bonuses(self)
 370        general_morale_regen = bonuses.get('morale_regen', 0)
 371        
 372        total_morale_regen = base_morale_regen + general_morale_regen
 373        self.stats.stats.morale = min(self.stats.stats.max_morale, self.stats.stats.morale + total_morale_regen)
 374        self.stats.stats.current_cohesion = min(
 375            self.stats.stats.max_cohesion,
 376            self.stats.stats.current_cohesion + base_cohesion_regen
 377        )
 378        
 379        # Check if routing unit recovers enough morale to stop routing
 380        self.has_rallied_this_turn = False
 381        
 382        # Check if unit has rallies remaining (Militia=0, Regular=1, Elite=3)
 383        can_rally = self.times_routed <= self.quality.max_rallies
 384        
 385        if (self.is_routing and can_rally and
 386                self.stats.stats.morale >= CombatConfig.RALLY_MORALE_THRESHOLD and
 387                self.stats.stats.current_cohesion >= CombatConfig.RALLY_COHESION_THRESHOLD):
 388            # Additional rally conditions for realism
 389            rally_chance = CombatConfig.RALLY_BASE_CHANCE
 390            
 391            # Better chance to rally with higher morale
 392            if self.stats.stats.morale >= CombatConfig.RALLY_HIGH_MORALE_THRESHOLD:
 393                rally_chance = CombatConfig.RALLY_HIGH_MORALE_CHANCE
 394
 395            if self.stats.stats.current_cohesion >= CombatConfig.RALLY_HIGH_COHESION_THRESHOLD:
 396                rally_chance *= CombatConfig.RALLY_HIGH_COHESION_MULTIPLIER
 397                
 398            # Reduced chance if heavily damaged
 399            soldier_ratio = self.soldiers / self.max_soldiers
 400            if soldier_ratio < 0.5:  # Less than 50% soldiers remaining
 401                rally_chance *= CombatConfig.RALLY_LOW_STRENGTH_MULTIPLIER
 402                
 403            import random
 404            if random.random() < rally_chance:
 405                self.is_routing = False
 406                self.has_rallied_this_turn = True
 407            
 408        # Reset temporary combat modifiers
 409        self.temp_damage_multiplier = 1.0
 410        self.temp_vulnerability = 1.0
 411        
 412        # Reset attack counter for new turn
 413        self.attacks_this_turn = 0
 414        
 415        # Update generals
 416        self.generals.on_turn_end()

 578    def attempt_breakaway(self, enemy_unit, game_state) -> Dict[str, Any]:
 579        """Attempt to break away from combat"""
 580        if not self.can_break_away_from(enemy_unit):
 581            return {'success': False, 'reason': 'Cannot break away'}
 582            
 583        # Calculate breakaway chance
 584        breakaway_chance = CombatConfig.get_breakaway_chance(
 585            enemy_unit.unit_class.value, 
 586            self.unit_class.value
 587        )
 588        
 589        # Roll for success
 590        import random
 591        roll = random.randint(1, 100)
 592        success = roll <= breakaway_chance
 593        
 594        # Consume AP
 595        self.action_points -= CombatConfig.BREAKAWAY_AP_COST
 596        
 597        if success:
 598            # Successful breakaway
 599            self.is_engaged_in_combat = False
 600            self.engaged_with = None
 601            enemy_unit.is_engaged_in_combat = False
 602            enemy_unit.engaged_with = None
 603            
 604            return {
 605                'success': True,
 606                'message': f'{self.name} successfully broke away from combat!',
 607                'opportunity_attack': True  # Enemy gets opportunity attack
 608            }
 609        else:
 610            # Failed breakaway
 611            self.morale = max(0, self.morale - CombatConfig.FAILED_BREAKAWAY_MORALE_LOSS)
 612            return {
 613                'success': False,
 614                'message': f'{self.name} failed to break away and lost morale!',
 615                'morale_loss': CombatConfig.FAILED_BREAKAWAY_MORALE_LOSS
