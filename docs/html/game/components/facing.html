 42    def update_facing_from_movement(self, from_x: int, from_y: int, to_x: int, to_y: int):
 43        """Update facing based on movement direction (Odd-R Flat-Top Layout)"""
 44        dx = to_x - from_x
 45        dy = to_y - from_y
 46        
 47        if dx == 0 and dy == 0:
 48            return  # No movement
 49            
 50        if dy == 0:
 51            # Pure horizontal movement
 52            if dx > 0:
 53                self.facing = FacingDirection.EAST
 54            else:
 55                self.facing = FacingDirection.WEST
 56        elif dy > 0:
 57            # Moving South
 58            if dx > 0:
 59                self.facing = FacingDirection.SOUTH_EAST
 60            elif dx < 0:
 61                self.facing = FacingDirection.SOUTH_WEST
 62            else:
 63                # dx == 0, depends on row parity
 64                if from_y % 2 == 0: # Even row, pushed left, so dx=0 is Right/East-ish
 65                    self.facing = FacingDirection.SOUTH_EAST
 66                else: # Odd row, pushed right, so dx=0 is Left/West-ish
 67                    self.facing = FacingDirection.SOUTH_WEST
 68        else:
 69            # Moving North
 70            if dx > 0:
 71                self.facing = FacingDirection.NORTH_EAST
 72            elif dx < 0:
 73                self.facing = FacingDirection.NORTH_WEST
 74            else:
 75                # dx == 0
 76                if from_y % 2 == 0: # Even row
 77                    self.facing = FacingDirection.NORTH_EAST
 78                else: # Odd row
 79                    self.facing = FacingDirection.NORTH_WEST
 80                
 81        self._last_move_direction = (dx, dy)

 83    def get_attack_angle(self, attacker_x: int, attacker_y: int, 
 84                        defender_x: int, defender_y: int) -> AttackAngle:
 85        """Determine if an attack is frontal, rear, or flank"""
 86        # Calculate relative position of attacker from defender's perspective
 87        dx = attacker_x - defender_x
 88        dy = attacker_y - defender_y
 89        
 90        # Convert to angle (0-360 degrees)
 91        angle_rad = math.atan2(dy, dx)
 92        angle_deg = (math.degrees(angle_rad) + 360) % 360
 93        
 94        # Get facing angle
 95        facing_angles = {
 96            FacingDirection.EAST: 0,
 97            FacingDirection.SOUTH_EAST: 60,
 98            FacingDirection.SOUTH_WEST: 120,
 99            FacingDirection.WEST: 180,
100            FacingDirection.NORTH_WEST: 240,
101            FacingDirection.NORTH_EAST: 300
102        }
103        
104        facing_angle = facing_angles[self.facing]
105        
106        # Calculate relative angle
107        relative_angle = (angle_deg - facing_angle + 360) % 360
108        
109        # Determine attack type based on relative angle
110        # Front arc: 300-360 and 0-60 degrees (120 degree arc)
111        # Rear arc: 120-240 degrees (120 degree arc)
112        # Flank arcs: 60-120 and 240-300 degrees (60 degrees each)
113        
114        if relative_angle <= 60 or relative_angle >= 300:
115            return AttackAngle(
116                is_frontal=True, 
117                is_rear=False, 
118                is_flank=False,
119                angle_degrees=relative_angle,
120                description="Frontal attack"
121            )
122        elif 120 <= relative_angle <= 240:
123            return AttackAngle(
124                is_frontal=False, 
125                is_rear=True, 
126                is_flank=False,
127                angle_degrees=relative_angle,
128                description="Rear attack"
129            )
130        else:
131            return AttackAngle(
132                is_frontal=False, 
133                is_rear=False, 
134                is_flank=True,
135                angle_degrees=relative_angle,
136                description="Flank attack"
137            )

166    def check_routing_chance(self, attack_angle: AttackAngle, 
167                           current_morale: float, casualties_percent: float) -> bool:
168        """Check if unit should route based on attack angle and casualties"""
169        import random
170        
171        # Base routing chance
172        base_chance = 0
173        
174        # Rear attacks have high routing chance
175        if attack_angle.is_rear:
176            if casualties_percent > 0.3:  # Lost 30% in one attack
177                base_chance = 60
178            elif casualties_percent > 0.2:  # Lost 20% in one attack
179                base_chance = 40
180            elif casualties_percent > 0.1:  # Lost 10% in one attack
181                base_chance = 20
182                
183            # Low morale increases chance
184            if current_morale < 40:
185                base_chance += 20
186            elif current_morale < 60:
187                base_chance += 10
188                
189        # Flank attacks have moderate routing chance
190        elif attack_angle.is_flank:
191            if casualties_percent > 0.4:  # Lost 40% in one attack
192                base_chance = 30
193            elif casualties_percent > 0.25:  # Lost 25% in one attack
194                base_chance = 15
195                
196            # Low morale increases chance
197            if current_morale < 30:
198                base_chance += 15
199                
200        # Even frontal attacks can cause routing with heavy casualties
201        else:
202            if casualties_percent > 0.5 and current_morale < 40:
203                base_chance = 20
204                
205        return random.randint(1, 100) <= base_chance
