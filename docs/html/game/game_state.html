 331    def _find_valid_position_near(self, x: int, y: int, max_distance: int = 3) -> Optional[tuple[int, int]]:
 332        """Find a valid position near the given position where terrain is passable.
 333        
 334        Returns a tuple (x, y) of a valid position, or None if no valid position found.
 335        """
 336        # First check if the original position is valid
 337        if self._is_position_valid_for_placement(x, y):
 338            return (x, y)
 339            
 340        # Search in expanding rectangles for simplicity
 341        for distance in range(1, max_distance + 1):
 342            # Check positions in a square around the original position
 343            positions_to_check = []
 344            
 345            # Top and bottom rows
 346            for dx in range(-distance, distance + 1):
 347                positions_to_check.append((x + dx, y - distance))
 348                positions_to_check.append((x + dx, y + distance))
 349            
 350            # Left and right columns (excluding corners already checked)
 351            for dy in range(-distance + 1, distance):
 352                positions_to_check.append((x - distance, y + dy))
 353                positions_to_check.append((x + distance, y + dy))
 354            
 355            # Check each position
 356            for new_x, new_y in positions_to_check:
 357                if (0 <= new_x < self.board_width and 
 358                    0 <= new_y < self.board_height and
 359                    self._is_position_valid_for_placement(new_x, new_y)):
 360                    return (new_x, new_y)
 361                    
 362        return None

 364    def _is_position_valid_for_placement(self, x: int, y: int) -> bool:
 365        """Check if a position is valid for placing a unit.
 366        
 367        Position is valid if:
 368        - Within board bounds
 369        - Terrain is passable
 370        - No other unit at this position
 371        - Not on a castle
 372        """
 373        # Check bounds
 374        if not (0 <= x < self.board_width and 0 <= y < self.board_height):
 375            return False
 376            
 377        # Check terrain
 378        if self.terrain_map:
 379            terrain = self.terrain_map.get_terrain(x, y)
 380            if terrain and not terrain.passable:
 381                return False
 382        
 383        # Check for other units
 384        for knight in self.knights:
 385            if knight.x == x and knight.y == y:
 386                return False
 387                
 388        # Check for castles
 389        for castle in self.castles:
 390            if hasattr(castle, 'contains_position') and castle.contains_position(x, y):
 391                return False
 392                
 393        return True

 920                        self.add_message(f"Castle archers fire! {total_damage} total damage dealt.")
 921        
 922        # Before switching players, clear OLD enemy paths (from 2 turns ago)
 923        # but keep current player's paths so next player can see them

 925        new_movement_history = {}
 926        
 927        for unit_id, path in self.movement_history.items():
 928            # Find the unit

 931                if id(knight) == unit_id:
 932                    unit = knight
 933                    break
 934            
 935            # Keep paths from the current turn (current player who just moved)
 936            # Clear paths from the next player (their old paths from last turn)
 937            if unit and unit.player_id == self.current_player:
 938                new_movement_history[unit_id] = path
 939            elif unit and unit.player_id == next_player:
 940                pass  # Clear old paths
 941

 943        
 944        self.current_player = 2 if self.current_player == 1 else 1

 946            self.turn_number += 1

 949

 951        self.fog_of_war.update_player_visibility(self, self.current_player)

 332    def _find_valid_position_near(self, x: int, y: int, max_distance: int = 3) -> Optional[tuple[int, int]]:
 333        """Find a valid position near the given position where terrain is passable.
 334        
 335        Returns a tuple (x, y) of a valid position, or None if no valid position found.
 336        """
 337        # First check if the original position is valid
 338        if self._is_position_valid_for_placement(x, y):
 339            return (x, y)
 340            
 341        # Search in expanding rectangles for simplicity
 342        for distance in range(1, max_distance + 1):
 343            # Check positions in a square around the original position
 344            positions_to_check = []
 345            
 346            # Top and bottom rows
 347            for dx in range(-distance, distance + 1):
 348                positions_to_check.append((x + dx, y - distance))
 349                positions_to_check.append((x + dx, y + distance))
 350            
 351            # Left and right columns (excluding corners already checked)
 352            for dy in range(-distance + 1, distance):
 353                positions_to_check.append((x - distance, y + dy))
 354                positions_to_check.append((x + distance, y + dy))
 355            
 356            # Check each position
 357            for new_x, new_y in positions_to_check:
 358                if (0 <= new_x < self.board_width and 
 359                    0 <= new_y < self.board_height and
 360                    self._is_position_valid_for_placement(new_x, new_y)):
 361                    return (new_x, new_y)
 362                    
 363        return None

 365    def _is_position_valid_for_placement(self, x: int, y: int) -> bool:
 366        """Check if a position is valid for placing a unit.
 367        
 368        Position is valid if:
 369        - Within board bounds
 370        - Terrain is passable
 371        - No other unit at this position
 372        - Not on a castle
 373        """
 374        # Check bounds
 375        if not (0 <= x < self.board_width and 0 <= y < self.board_height):
 376            return False
 377            
 378        # Check terrain
 379        if self.terrain_map:
 380            terrain = self.terrain_map.get_terrain(x, y)
 381            if terrain and not terrain.passable:
 382                return False
 383        
 384        # Check for other units
 385        for knight in self.knights:
 386            if knight.x == x and knight.y == y:
 387                return False
 388                
 389        # Check for castles
 390        for castle in self.castles:
 391            if hasattr(castle, 'contains_position') and castle.contains_position(x, y):
 392                return False
 393                
 394        return True

 603                self.selected_knight.consume_move_ap()
 604                self.pending_positions[id(self.selected_knight)] = (tile_x, tile_y)

 684            target = self.get_knight_at(tile_x, tile_y)
 685            if not target or target.player_id == self.current_player:
 686                return False
 687
 688            # Use attack behavior if available (which includes terrain-based AP costs and line of sight)
 689            if hasattr(self.selected_knight, 'behaviors') and 'attack' in self.selected_knight.behaviors:
 690                    attack_behavior = self.selected_knight.behaviors['attack']

 693                    if target not in valid_targets:
 694                        return False  # Attack blocked - feedback will be provided by input handler
 695                    result = self.selected_knight.behaviors['attack'].execute(self.selected_knight, self, target)
 696                    if result['success']:
 697                        damage = result['damage']
 698                        counter_damage = result.get('counter_damage', 0)
 699                        attack_angle = result.get('attack_angle', None)

 708                                             game_state=self)
 709                        self.animation_coordinator.animation_manager.add_animation(anim)
 710                    else:
 711                        return False
 712            else:
 713                # Fallback to old method with terrain-based AP cost
 714                # Proceed with attack
 715                self.selected_knight.has_attacked = True
 716                
 717                damage = max(1, self.selected_knight.soldiers // 10)  # Fallback damage
 718                
 719                # Apply damage
 720                target.take_casualties(damage, self)
 721                
 722                # Calculate terrain-based AP cost
 723                ap_cost = 3  # Base cost
 724                if hasattr(self.selected_knight, 'unit_class'):
 725                    if self.selected_knight.unit_class == KnightClass.WARRIOR:

 730                        ap_cost = 3

 732                        ap_cost = 2
 733

 735                if self.terrain_map:
 736                    target_terrain = self.terrain_map.get_terrain(target.x, target.y)
 737                    if target_terrain:
 738                        terrain_movement_cost = target_terrain.movement_cost

 894    def end_turn(self):
 895        # Check for rallied units before ending turn
 896        rallied_units = []
 897        for knight in self.knights:
 898            if knight.player_id == self.current_player:
 899                knight.end_turn()
 900                # Check if unit rallied this turn
 901                if hasattr(knight, 'has_rallied_this_turn') and knight.has_rallied_this_turn:
 902                    rallied_units.append(knight)
 903        
 904        # Add rally messages
 905        for knight in rallied_units:
 906            self.add_message(f"{knight.name} rallies and stops routing!", priority=2)
 907        
 908        for castle in self.castles:
 909            if castle.player_id == self.current_player:
 910                enemies = castle.get_enemies_in_range(self.knights)
 911                if enemies:
 912                    damages = castle.shoot_arrows(enemies)
 913                    
 914                    # Create arrow animation - animation will apply damage when arrows hit
 915                    if damages:
 916                        anim = ArrowAnimation(castle, enemies, [d[1] for d in damages])
 917                        self.animation_coordinator.animation_manager.add_animation(anim)
 918                        
 919                        # Add message for castle archer volleys
 920                        total_damage = sum(d[1] for d in damages)
 921                        self.add_message(f"Castle archers fire! {total_damage} total damage dealt.")
 922        
 923        # Before switching players, clear OLD enemy paths (from 2 turns ago)
 924        # but keep current player's paths so next player can see them
 925        next_player = 2 if self.current_player == 1 else 1
 926        new_movement_history = {}
 927        
 928        for unit_id, path in self.movement_history.items():
 929            # Find the unit
 930            unit = None
 931            for knight in self.knights:
 932                if id(knight) == unit_id:
 933                    unit = knight
 934                    break
 935            
 936            # Keep paths from the current turn (current player who just moved)
 937            # Clear paths from the next player (their old paths from last turn)
 938            if unit and unit.player_id == self.current_player:
 939                new_movement_history[unit_id] = path
 940            elif unit and unit.player_id == next_player:
 941                pass  # Clear old paths
 942        
 943        self.movement_history = new_movement_history
 944        
 945        self.current_player = 2 if self.current_player == 1 else 1
 946        if self.current_player == 1:
 947            self.turn_number += 1
 948        
 949        self.deselect_knight()
 950        
 951        # Update fog of war ONLY for the new current player
 952        self.fog_of_war.update_player_visibility(self, self.current_player)
 953        
 954        for castle in self.castles:
 955            if castle.player_id == self.current_player:
 956                castle.end_turn()
