  9    """Manages and executes end-turn steps in the correct order."""
 10    
 11    def __init__(self):
 12        self._steps: List[EndTurnStep] = []
 13        self._step_map: Dict[str, EndTurnStep] = {}
 14    
 15    def register_step(self, step: EndTurnStep) -> None:
 16        """Register a new end-turn step."""
 17        if step.name in self._step_map:
 18            raise ValueError(f"Step with name '{step.name}' already registered")
 19        
 20        self._steps.append(step)
 21        self._step_map[step.name] = step
 22        logger.info(f"Registered end-turn step: {step.name} (priority: {step.priority})")
 23    
 24    def unregister_step(self, step_name: str) -> None:
 25        """Remove a step from the processor."""
 26        if step_name in self._step_map:
 27            step = self._step_map[step_name]
 28            self._steps.remove(step)
 29            del self._step_map[step_name]
 30            logger.info(f"Unregistered end-turn step: {step_name}")
 31    
 32    def get_execution_order(self) -> List[EndTurnStep]:
 33        """
 34        Get the steps in execution order, respecting priorities and dependencies.
 35        Uses topological sort to handle dependencies correctly.
 36        """
 37        # First sort by priority
 38        sorted_steps = sorted(self._steps, key=lambda s: s.priority)
 39        
 40        # Build dependency graph
 41        dependencies: Dict[str, Set[str]] = {}
 42        for step in sorted_steps:
 43            dependencies[step.name] = set(step.dependencies)
 44        
 45        # Topological sort within each priority level
 46        result = []
 47        visited = set()
 48        temp_visited = set()
 49        
 50        def visit(step_name: str):
 51            if step_name in temp_visited:
 52                raise ValueError(f"Circular dependency detected involving {step_name}")
 53            if step_name in visited:
 54                return
 55            
 56            temp_visited.add(step_name)
 57            
 58            # Visit dependencies first
 59            step = self._step_map.get(step_name)
 60            if step:
 61                for dep in step.dependencies:
 62                    if dep in self._step_map:
 63                        visit(dep)
 64            
 65            temp_visited.remove(step_name)
 66            visited.add(step_name)
 67            
 68            if step:
 69                result.append(step)
 70        
 71        # Process all steps
 72        for step in sorted_steps:
 73            visit(step.name)
 74        
 75        return result
 76    
 77    def execute(self, campaign_state, current_country_id: int, turn_number: int) -> EndTurnContext:
 78        """
 79        Execute all registered end-turn steps in order.
 80        
 81        Args:
 82            campaign_state: The current campaign state
 83            current_country_id: ID of the country whose turn is ending
 84            turn_number: Current turn number
 85            
 86        Returns:
 87            The context with all step results
 88        """
 89        context = EndTurnContext(
 90            campaign_state=campaign_state,
 91            current_country_id=current_country_id,
 92            turn_number=turn_number,
 93            step_results={}
 94        )
 95        
 96        execution_order = self.get_execution_order()
 97        logger.info(f"Executing {len(execution_order)} end-turn steps for country {current_country_id}")
 98        
 99        for step in execution_order:
100            try:
101                # Check dependencies
102                if not step.validate_dependencies(context):
103                    logger.error(f"Step {step.name} missing dependencies: {step.dependencies}")
104                    continue
105                
106                # Check if step should execute
107                if not step.should_execute(context):
108                    logger.debug(f"Skipping step {step.name} (should_execute returned False)")
109                    continue
110                
111                logger.debug(f"Executing step: {step.name}")
112                result = step.execute(context)
113                
114                if result is not None:
115                    context.set_result(step.name, result)
116                    
117            except Exception as e:
118                logger.error(f"Error executing step {step.name}: {e}", exc_info=True)
119                # Continue with other steps even if one fails
120        
121        logger.info("End-turn processing complete")
122        return context
