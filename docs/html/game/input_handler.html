 59                                reason = attack_behavior.get_attack_blocked_reason(game_state.selected_knight, target, game_state)

 61                            else:
 62                                game_state.add_message("Cannot attack target", priority=1)

 66                            game_state.add_message("Cannot attack - no unit selected", priority=1)

 79                else:
 80                    # Use hex layout to convert click to hex coordinates

200    def zoom_in(self, game_state):
201        """Zoom in the view"""
202        if not hasattr(game_state, 'camera_manager'):
203            raise ValueError("game_state.camera_manager is required for zooming")
204        if game_state.camera_manager.zoom_in():
205            self.update_zoom(game_state)

207    def zoom_out(self, game_state):
208        """Zoom out the view"""
209        if not hasattr(game_state, 'camera_manager'):
210            raise ValueError("game_state.camera_manager is required for zooming")
211        if game_state.camera_manager.zoom_out():
212            self.update_zoom(game_state)

214    def update_zoom(self, game_state):
215        """Update hex grid and layout with new zoom level (original working implementation)"""
216        # Get zoom level from camera manager if available, otherwise use local zoom
217        if not hasattr(game_state, 'camera_manager'):
218            raise ValueError("game_state.camera_manager is required for zoom updates")
219        zoom_level = game_state.camera_manager.zoom_level
220        
221        # Scale hex size based on zoom level
222        new_hex_size = int(36 * zoom_level)
223        self.hex_grid = HexGrid(hex_size=new_hex_size)
224        self.hex_layout = HexLayout(hex_size=new_hex_size)
225        
226        # Update game_state's hex_layout to ensure coordinate conversion consistency
227        if hasattr(game_state, 'hex_layout'):
228            game_state.hex_layout = self.hex_layout
229            
230        # Update renderer's hex_layout if available to maintain consistency
231        if hasattr(game_state, 'renderer') and hasattr(game_state.renderer, 'hex_layout'):
232            game_state.renderer.hex_layout = self.hex_layout
233            game_state.renderer.hex_grid = self.hex_grid
234            # Update all sub-renderers if using the new rendering system
235            if hasattr(game_state.renderer, 'terrain_renderer'):
236                game_state.renderer.terrain_renderer.hex_layout = self.hex_layout
237                game_state.renderer.terrain_renderer.hex_grid = self.hex_grid
238            if hasattr(game_state.renderer, 'unit_renderer'):
239                game_state.renderer.unit_renderer.hex_layout = self.hex_layout
240            if hasattr(game_state.renderer, 'effect_renderer'):
241                game_state.renderer.effect_renderer.hex_layout = self.hex_layout
242
243        # Update camera manager world bounds so the camera can pan to the
244        # board edges even at high zoom levels. Calculate approximate board
245        # pixel dimensions from the hex layout and scale by 2 to match the
246        # camera manager's internal bounds logic.
247        if hasattr(game_state, 'camera_manager'):
248            hl = self.hex_layout
249            board_pixel_width = ((game_state.board_width - 1) * hl.col_spacing +
250                                 hl.hex_width + (hl.row_offset if game_state.board_height > 1 else 0))
251            board_pixel_height = ((game_state.board_height - 1) * hl.row_spacing +
252                                  hl.hex_height)
253
254            world_width = int(board_pixel_width * 2)
255            world_height = int(board_pixel_height * 2)
256            game_state.camera_manager.set_world_bounds(world_width, world_height)
257            # Re-apply current camera position so it remains clamped to the new bounds
258            game_state.camera_manager.set_camera_position(
259                game_state.camera_manager.camera_x,
260                game_state.camera_manager.camera_y
261            )

263    def screen_to_world_zoom_aware(self, screen_x, screen_y, game_state):
264        """Convert screen coordinates to world coordinates accounting for zoom"""
265        # Get base world coordinates
266        world_x, world_y = game_state.screen_to_world(screen_x, screen_y)
267        return world_x, world_y

270        """Get current zoom level"""
271        return self.zoom_level
