 19    def __init__(self, attack_range: int = 1):

 23    def can_execute(self, unit, game_state) -> bool:
 24        """Check if unit can attack"""
 25        base_cost = self.get_ap_cost(unit)  # Get base cost without terrain penalty
 26        
 27        # Basic requirements - remove has_acted check to allow multiple attacks
 28        if unit.action_points < base_cost:
 29            return False
 30            
 31        # Morale requirement: Need 50%+ morale to attack (except first attack)
 32        if not hasattr(unit, 'attacks_this_turn'):
 33            unit.attacks_this_turn = 0
 34            
 35        if unit.attacks_this_turn > 0 and unit.morale < 50:
 36            return False
 37            
 38        return True

 40    def get_ap_cost(self, unit=None, target=None, game_state=None) -> int:
 41        """Calculate AP cost for attack based on unit type, target terrain, and distance"""
 42        base_cost = 3  # Default
 43
 44        if unit and hasattr(unit, 'unit_class'):
 45            from game.combat_config import CombatConfig
 46            base_cost = CombatConfig.get_attack_ap_cost(unit.unit_class.value)
 47        
 48        # Add terrain-based attack cost if target and game_state are provided
 49        if target and game_state and hasattr(game_state, 'terrain_map') and game_state.terrain_map:
 50            target_terrain = game_state.terrain_map.get_terrain(target.x, target.y)
 51            if target_terrain:
 52                # Get the base terrain movement cost (without unit-specific modifiers)
 53                terrain_movement_cost = target_terrain.movement_cost
 54                
 55                # Apply terrain penalty: twice the additional movement cost as extra AP
 56                if terrain_movement_cost > 1.0:  # Only apply penalty for difficult terrain
 57                    terrain_penalty = int((terrain_movement_cost - 1.0) * 2)
 58                    base_cost += terrain_penalty
 59                
 60        return base_cost

 62    def determine_combat_mode(self, attacker, target, distance: int) -> CombatMode:
 63        """Determine the combat mode based on units and distance"""
 64        # Ranged combat when attacking from distance
 65        if distance > 1:
 66            return CombatMode.RANGED
 67            
 68        # Melee combat at close range
 69        if distance == 1:
 70            # Special case: cavalry charging (could be expanded with charge mechanic)
 71            if attacker.unit_class == KnightClass.CAVALRY and not hasattr(attacker, 'has_charged'):
 72                return CombatMode.CHARGE
 73                
 74            # Standard melee combat
 75            return CombatMode.MELEE
 76            
 77        # Should not reach here, but default to melee
 78        return CombatMode.MELEE

 80    def execute(self, unit, game_state, target) -> Dict[str, Any]:
 81        """Execute attack against target"""
 82        if not self.can_execute(unit, game_state):
 83            return {'success': False, 'reason': 'Cannot attack'}
 84            
 85        # Check range (Chebyshev distance for grid movement)
 86        dx = abs(unit.x - target.x)
 87        dy = abs(unit.y - target.y)
 88        distance = max(dx, dy)  # Chebyshev distance allows diagonals
 89        
 90        if distance > self.attack_range:
 91            return {'success': False, 'reason': 'Target out of range'}
 92            
 93        # Check target is enemy
 94        if target.player_id == unit.player_id:
 95            return {'success': False, 'reason': 'Cannot attack friendly units'}
 96            
 97        # Determine combat mode
 98        combat_mode = self.determine_combat_mode(unit, target, distance)
 99        
100        # Get terrain modifiers
101        attacker_terrain = game_state.terrain_map.get_terrain(unit.x, unit.y)
102        target_terrain = game_state.terrain_map.get_terrain(target.x, target.y)
103        
104        # Calculate damage based on combat mode
105        damage = self.calculate_damage(unit, target, attacker_terrain, target_terrain, combat_mode)
106        
107        # Calculate counter damage based on combat mode
108        counter_damage = 0
109        if combat_mode == CombatMode.MELEE:
110            # Full counter-attack in melee
111            counter_damage = self.calculate_counter_damage(target, unit, target_terrain, attacker_terrain)
112        elif combat_mode == CombatMode.SKIRMISH:
113            # Reduced counter-attack for skirmish
114            counter_damage = int(self.calculate_counter_damage(target, unit, target_terrain, attacker_terrain) * 0.5)
115        elif combat_mode == CombatMode.CHARGE:
116            # Cavalry charges can receive counter-attack but at reduced effectiveness
117            counter_damage = int(self.calculate_counter_damage(target, unit, target_terrain, attacker_terrain) * 0.75)
118        # RANGED mode has no counter-attack (counter_damage remains 0)
119            
120        # Consume AP (including terrain penalty)
121        ap_cost = self.get_ap_cost(unit, target, game_state)
122        unit.action_points -= ap_cost
123        # Mark unit as having acted this turn
124        unit.has_acted = True
125        
126        # Track attacks this turn and apply progressive morale loss
127        if not hasattr(unit, 'attacks_this_turn'):
128            unit.attacks_this_turn = 0
129            
130        unit.attacks_this_turn += 1
131        
132        # Apply morale loss for multiple attacks (combat fatigue)
133        if unit.attacks_this_turn > 1:
134            # Progressive morale loss: 5% for second attack, 10% for third, etc.
135            morale_loss = unit.attacks_this_turn * 5
136            # Apply to base morale stats to avoid general bonus interference
137            unit.stats.stats.morale = max(0, unit.stats.stats.morale - morale_loss)
138        
139        # Mark units as engaged in combat (only for melee/close combat)
140        if combat_mode in [CombatMode.MELEE, CombatMode.CHARGE]:
141            unit.is_engaged_in_combat = True
142            unit.engaged_with = target
143            target.is_engaged_in_combat = True
144            target.engaged_with = unit
145        # Ranged attacks do not engage units in combat
146        
147        # Check attack angle for additional effects
148        attack_angle = None
149        extra_morale_penalty = 0
150        should_check_routing = False
151        
152        if hasattr(target, 'facing'):
153            attack_angle = target.facing.get_attack_angle(unit.x, unit.y, target.x, target.y)
154            extra_morale_penalty = target.facing.get_morale_penalty(attack_angle)
155            
156            # Check for routing on rear/flank attacks
157            if attack_angle.is_rear or attack_angle.is_flank:
158                should_check_routing = True
159        
160        return {
161            'success': True,
162            'damage': damage,
163            'counter_damage': counter_damage,
164            'animation': 'attack',
165            'target': target,
166            'attack_angle': attack_angle,
167            'extra_morale_penalty': extra_morale_penalty,
168            'should_check_routing': should_check_routing,
169            'combat_mode': combat_mode
170        }

172    def calculate_damage(self, attacker, target, attacker_terrain=None, target_terrain=None, combat_mode=None) -> int:
173        """Calculate damage to target based on combat mode"""
174        # Get effective soldiers
175        attacking_soldiers = attacker.stats.get_effective_soldiers(attacker_terrain)
176        base_damage = attacking_soldiers * attacker.stats.stats.attack_per_soldier
177        
178        # Apply combat mode modifiers
179        if combat_mode:
180            if combat_mode == CombatMode.CHARGE:
181                # Cavalry charges deal bonus damage
182                base_damage *= 1.3
183            elif combat_mode == CombatMode.SKIRMISH:
184                # Skirmish attacks deal reduced damage
185                base_damage *= 0.8
186            # RANGED and MELEE use normal damage
187        
188        # Apply terrain combat modifier (but separate from height advantage)
189        if attacker_terrain:
190            terrain_modifier = attacker_terrain.get_combat_modifier_for_unit(attacker.unit_class)
191            base_damage *= terrain_modifier
192            
193        # Apply morale modifier
194        base_damage *= (attacker.morale / 100)  # Use property that includes general bonuses
195        
196        # Apply disrupted penalty
197        if hasattr(attacker, 'is_disrupted') and attacker.is_disrupted:
198            base_damage *= 0.5  # 50% damage penalty when disrupted
199        
200        # Apply general damage bonuses
201        if hasattr(attacker, 'get_damage_modifier'):
202            base_damage *= attacker.get_damage_modifier()
203            
204        # Apply facing modifier
205        if hasattr(target, 'facing'):
206            attack_angle = target.facing.get_attack_angle(attacker.x, attacker.y, target.x, target.y)
207            facing_modifier = target.facing.get_damage_modifier(attack_angle)
208            base_damage *= facing_modifier
209            
210        # Apply height advantage/disadvantage for ranged attacks
211        if self.attack_range > 1 and attacker_terrain and target_terrain:
212            # Check if attacker is shooting from lower ground to higher ground
213            attacker_on_hills = attacker_terrain.type.value.lower() == 'hills'
214            target_on_hills = target_terrain.type.value.lower() == 'hills'
215            
216            if not attacker_on_hills and target_on_hills:
217                # Shooting uphill - 50% damage penalty
218                base_damage = int(base_damage * 0.5)
219            elif attacker_on_hills and not target_on_hills:
220                # Shooting downhill - 50% damage bonus
221                base_damage = int(base_damage * 1.5)
222            
223        # Calculate defense with general bonuses
224        target_defense = target.stats.stats.base_defense
225        if hasattr(target, 'generals'):
226            defense_bonuses = target.generals.get_all_passive_bonuses(target)
227            target_defense *= (1 + defense_bonuses.get('defense_bonus', 0))
228            
229        if target_terrain:
230            target_defense += target_terrain.defense_bonus
231            
232        # Apply disrupted penalty to defense
233        if hasattr(target, 'is_disrupted') and target.is_disrupted:
234            target_defense *= 0.5  # 50% defense penalty when disrupted
235            
236        # Special case: garrisoned units
237        if hasattr(target, 'is_garrisoned') and target.is_garrisoned:
238            target_defense += 20
239            
240        # Calculate casualties
241        damage_ratio = base_damage / (base_damage + target_defense)
242        base_casualties = int(damage_ratio * attacking_soldiers * 0.25)
243        
244        # Unit matchup bonuses
245        if attacker.unit_class == KnightClass.CAVALRY and target.unit_class == KnightClass.ARCHER:
246            base_casualties = int(base_casualties * 1.5)
247        elif attacker.unit_class == KnightClass.ARCHER and target.unit_class == KnightClass.WARRIOR:
248            base_casualties = int(base_casualties * 0.8)
249            
250        return min(base_casualties, target.stats.stats.current_soldiers)
251        
252    def calculate_counter_damage(self, defender, attacker, defender_terrain=None, attacker_terrain=None) -> int:
253        """Calculate counter damage in melee"""
254        # Archers don't counter in melee
255        if defender.unit_class == KnightClass.ARCHER:
256            return 0
257            
258        # Counter damage calculations
259        defending_soldiers = defender.stats.get_effective_soldiers(defender_terrain)
260        base_damage = defender.stats.stats.attack_per_soldier * defending_soldiers
261        
262        if defender_terrain:
263            base_damage *= defender_terrain.get_combat_modifier_for_unit(defender.unit_class)
264            
265        # Reduced morale effect for counter-attacks
266        base_damage *= (defender.stats.stats.morale / 200)
267        
268        # Calculate attacker's defense
269        attacker_defense = attacker.stats.stats.base_defense
270        if attacker_terrain:
271            attacker_defense += attacker_terrain.defense_bonus
272            
273        # Calculate casualties - counter attacks deal less
274        damage_ratio = base_damage / (base_damage + attacker_defense)
275        base_casualties = int(damage_ratio * defending_soldiers * 0.15)
276        
277        # Bonus if defender is warrior against cavalry
278        if defender.unit_class == KnightClass.WARRIOR and attacker.unit_class == KnightClass.CAVALRY:
279            base_casualties = int(base_casualties * 1.2)
280            
281        return min(base_casualties, attacker.stats.stats.current_soldiers)

283    def _is_valid_target(self, unit, target, game_state) -> bool:
284        """Check if a target is valid for attack"""
285        # Check fog of war visibility
286        if hasattr(game_state, 'fog_of_war') and game_state.current_player is not None:
287            visibility = game_state.fog_of_war.get_visibility_state(
288                game_state.current_player, target.x, target.y
289            )
290            # Only target units we can see
291            if visibility != VisibilityState.VISIBLE:
292                return False
293        
294        # Use Chebyshev distance to include diagonals
295        dx = abs(unit.x - target.x)
296        dy = abs(unit.y - target.y)
297        distance = max(dx, dy)
298        if distance > self.attack_range:
299            return False
300            
301        return True

303    def get_valid_targets(self, unit, game_state) -> list:
304        """Get list of valid attack targets"""
305        if not self.can_execute(unit, game_state):
306            return []
307            
308        targets = []
309        for other in game_state.knights:
310            if other.player_id != unit.player_id and self._is_valid_target(unit, other, game_state):
311                targets.append(other)
312                    
313        return targets

315    def get_attack_blocked_reason(self, unit, target, game_state) -> str:
316        """Get reason why an attack is blocked (for user feedback)"""
317        # Check basic requirements first
318        if not self.can_execute(unit, game_state):
319            base_cost = self.get_ap_cost(unit)
320            if unit.action_points < base_cost:
321                return "Not enough action points"
322            if hasattr(unit, 'attacks_this_turn') and unit.attacks_this_turn > 0 and unit.morale < 50:
323                return "Morale too low for additional attacks"
324        
325        # Check if it's an enemy
326        if target.player_id == unit.player_id:
327            return "Cannot attack friendly units"
328            
329        # Check visibility
330        if hasattr(game_state, 'fog_of_war') and game_state.current_player is not None:
331            visibility = game_state.fog_of_war.get_visibility_state(
332                game_state.current_player, target.x, target.y
333            )
334            if visibility != VisibilityState.VISIBLE:
335                return "Target not visible"
336        
337        # Check range
338        dx = abs(unit.x - target.x)
339        dy = abs(unit.y - target.y)
340        distance = max(dx, dy)
341        if distance > self.attack_range:
342            return f"Target out of range (distance: {distance}, max range: {self.attack_range})"
343            
344        return "Attack should be possible"
