 20class VisibilityState(Enum):
 21    """States for hex visibility."""
 22    HIDDEN = 0      # Never seen, no terrain info
 23    EXPLORED = 1    # Previously seen, terrain visible but grayed out
 24    PARTIAL = 2     # Can see terrain and that a unit exists, but not details
 25    VISIBLE = 3     # Full visibility of terrain and unit details

 65    def update_player_visibility(self, game_state, player_id: int):
 66        """Update visibility map for a specific player based on their units."""
 67        from game.entities.unit import Unit
 68        from game.entities.castle import Castle
 69        
 70        # Cache game_state for use in helper methods
 71        self._cached_game_state = game_state
 72        
 73        # First, downgrade all visible hexes to explored
 74        for coords in self.visibility_maps[player_id]:
 75            if self.visibility_maps[player_id][coords] == VisibilityState.VISIBLE:
 76                self.visibility_maps[player_id][coords] = VisibilityState.EXPLORED
 77            elif self.visibility_maps[player_id][coords] == VisibilityState.PARTIAL:
 78                self.visibility_maps[player_id][coords] = VisibilityState.EXPLORED
 79                
 80        # Get all units belonging to this player
 81        player_units = []
 82        for unit in game_state.units:
 83            if unit.player_id == player_id:
 84                player_units.append(unit)
 85                
 86        # Add castle vision if player has a castle
 87        castle_positions = []
 88        for castle in game_state.castles:
 89            if castle.player_id == player_id:
 90                # Castle has multiple hex footprint, use occupied tiles
 91                castle_positions.extend(castle.occupied_tiles)
 92                        
 93        # Calculate visibility from each unit
 94        for unit in player_units:
 95            vision_range = self._get_unit_vision_range(unit)
 96            # Check if unit has elevated vision
 97            vision_behavior = unit.get_behavior('VisionBehavior') if hasattr(unit, 'get_behavior') else None
 98            is_elevated = vision_behavior.is_elevated() if vision_behavior else False
 99            
100            visible_hexes = self._calculate_los_from_position(
101                game_state, 
102                (unit.x, unit.y), 
103                vision_range,
104                is_elevated
105            )
106            
107            # Update visibility states
108            for hex_coords, distance in visible_hexes.items():
109                current_state = self.visibility_maps[player_id][hex_coords]
110                
111                # Determine new visibility state based on distance
112                if distance <= self.vision_config.full_id_range:
113                    new_state = VisibilityState.VISIBLE
114                elif distance <= self.vision_config.partial_id_range:
115                    new_state = VisibilityState.PARTIAL
116                else:
117                    new_state = VisibilityState.EXPLORED
118                    
119                # Only upgrade visibility, never downgrade
120                if new_state.value > current_state.value:
121                    self.visibility_maps[player_id][hex_coords] = new_state
122                    
123        # Add castle vision (castles have fixed vision range)
124        for pos in castle_positions:
125            visible_hexes = self._calculate_los_from_position(
126                game_state,
127                pos,
128                4,  # Castles have good vision
129                True  # Castles are tall structures
130            )
131            
132            for hex_coords, distance in visible_hexes.items():
133                current_state = self.visibility_maps[player_id].get(hex_coords, VisibilityState.HIDDEN)
134                if distance <= 2:
135                    new_state = VisibilityState.VISIBLE
136                else:
137                    new_state = VisibilityState.PARTIAL
138                    
139                if new_state.value > current_state.value:
140                    self.visibility_maps[player_id][hex_coords] = new_state

142    def _get_unit_vision_range(self, unit) -> int:
143        """Get vision range for a unit using its vision behavior."""
144        # Get vision behavior if it exists
145        vision_behavior = unit.get_behavior('VisionBehavior') if hasattr(unit, 'get_behavior') else None
146        
147        if vision_behavior:
148            # Use the behavior to get vision range
149            terrain = None
150            if hasattr(unit, 'get_terrain'):
151                # Pass game_state to get_terrain if available
152                terrain = unit.get_terrain(self._cached_game_state if hasattr(self, '_cached_game_state') else None)
153            return vision_behavior.get_vision_range(terrain)
154        
155        # Fallback for units without vision behavior
156        return self.vision_config.default_range

198    def get_visibility_from_position(
199        self,
200        game_state,
201        unit,
202        origin: Tuple[int, int],
203        target: Tuple[int, int]
204    ) -> VisibilityState:
205        """Calculate visibility state for a target from a specific origin."""
206        if game_state is None:
207            raise ValueError("game_state is required to calculate visibility")
208        if unit is None:
209            raise ValueError("unit is required to calculate visibility")
210        if origin is None or target is None:
211            raise ValueError("origin and target are required to calculate visibility")
212
213        origin_x, origin_y = origin
214        target_x, target_y = target
215
216        if not self._is_valid_hex(origin_x, origin_y):
217            raise ValueError(f"Origin ({origin_x}, {origin_y}) is out of bounds")
218        if not self._is_valid_hex(target_x, target_y):
219            raise ValueError(f"Target ({target_x}, {target_y}) is out of bounds")
220
221        self._cached_game_state = game_state
222        vision_range = self._get_unit_vision_range(unit)
223        if vision_range <= 0:
224            return VisibilityState.HIDDEN
225
226        vision_behavior = unit.get_behavior('VisionBehavior') if hasattr(unit, 'get_behavior') else None
227        is_elevated = vision_behavior.is_elevated() if vision_behavior else False
228
229        visible_hexes = self._calculate_los_from_position(
230            game_state,
231            (origin_x, origin_y),
232            vision_range,
233            is_elevated
234        )
235
236        distance = visible_hexes.get((target_x, target_y))
237        if distance is None:
238            return VisibilityState.HIDDEN
239        if distance <= self.vision_config.full_id_range:
240            return VisibilityState.VISIBLE
241        if distance <= self.vision_config.partial_id_range:
242            return VisibilityState.PARTIAL
243        return VisibilityState.EXPLORED

245    def _calculate_los_from_position(self, game_state, origin: Tuple[int, int], 
246                                   max_range: int, is_elevated: bool = False) -> Dict[Tuple[int, int], int]:
247        """
248        Calculate line of sight from a position using shadow casting.
249        Returns dict of visible hex coordinates -> distance.
250        """
251        # Use the shadow caster for efficient visibility calculation
252        visible_hexes = self.shadowcaster.calculate_visible_hexes(
253            game_state, origin, max_range, is_elevated
254        )
255        
256        # Additional filtering: verify line of sight for edge cases
257        # The shadow caster gives us a good approximation, but we might want
258        # to double-check certain hexes, especially those partially in shadow
259        verified_hexes = {}
260        for hex_pos, distance in visible_hexes.items():
261            # For now, trust the shadow caster completely
262            # In a more complex implementation, we might verify edge cases
263            verified_hexes[hex_pos] = distance
264            
265        return verified_hexes

267    def _has_line_of_sight(self, game_state, origin: Tuple[int, int], 
268                          target: Tuple[int, int], is_elevated: bool) -> bool:
269        """Check if there's line of sight between two hexes."""
270        # Simple line of sight - check each hex along the path
271        hex_grid = HexGrid()
272        origin_hex = hex_grid.offset_to_axial(origin[0], origin[1])
273        target_hex = hex_grid.offset_to_axial(target[0], target[1])
274        
275        # Get line between hexes
276        line_hexes = self._get_line(origin_hex, target_hex)
277        
278        # Check each hex along the line for blocking
279        for i, hex_coord in enumerate(line_hexes[1:-1], 1):  # Skip origin and target
280            # Convert back to offset coordinates
281            hex_x, hex_y = hex_grid.axial_to_offset(hex_coord)
282            
283            # Check terrain blocking
284            terrain = game_state.terrain_map.get_terrain(hex_x, hex_y)
285            
286            if terrain:
287                terrain_type = terrain.type.value.lower()
288                
289                # Mountains always block line of sight (impassable high terrain)
290                if terrain_type == 'mountains':
291                    return False
292                
293                # High hills block unless viewer is elevated or on high ground
294                if terrain_type == 'high hills':
295                    origin_terrain = game_state.terrain_map.get_terrain(origin[0], origin[1])
296                    viewer_on_high_ground = origin_terrain and origin_terrain.type.value.lower() in ['hills', 'high hills', 'mountains']
297                    if not is_elevated and not viewer_on_high_ground:
298                        return False
299                
300                # Hills block unless viewer is elevated or on a hill
301                if terrain_type == 'hills':
302                    origin_terrain = game_state.terrain_map.get_terrain(origin[0], origin[1])
303                    # If viewer is on hills or is elevated (cavalry), they can see over hills
304                    viewer_on_hills = origin_terrain and origin_terrain.type.value.lower() in ['hills', 'high hills', 'mountains']
305                    if not is_elevated and not viewer_on_hills:
306                        return False
307            
308            # Check castle blocking
309            if hasattr(game_state, 'castles'):
310                for castle in game_state.castles:
311                    if hasattr(castle, 'contains_position') and castle.contains_position(hex_x, hex_y):
312                        # Castles are tall, but unless viewer is elevated or on a mountain, they block LOS
313                        origin_terrain = game_state.terrain_map.get_terrain(origin[0], origin[1])
314                        viewer_on_mountain = origin_terrain and origin_terrain.type.value.lower() == 'mountains'
315                        if not is_elevated and not viewer_on_mountain:
316                            return False
317                    
318            # Check unit blocking
319            blocking_unit = game_state.get_unit_at(hex_x, hex_y)
320            if blocking_unit:
321                # Check if unit blocks vision using its behavior
322                vision_behavior = blocking_unit.get_behavior('VisionBehavior') if hasattr(blocking_unit, 'get_behavior') else None
323                if vision_behavior and vision_behavior.blocks_vision():
324                    # Check if blocker is also elevated
325                    blocker_elevated = vision_behavior.is_elevated() if hasattr(vision_behavior, 'is_elevated') else False
326                    
327                    # Elevated viewers can see over non-elevated blockers
328                    if is_elevated and not blocker_elevated:
329                        continue
330                        
331                    # If both are elevated (e.g., cavalry viewing past cavalry), no blocking
332                    if is_elevated and blocker_elevated:
333                        continue
334                        
335                    # Otherwise check if target is behind the blocker
336                    if self._is_behind_blocker(origin, (hex_x, hex_y), target):
337                        return False
338                        
339        return True
340

342    def _is_behind_blocker(viewer: Tuple[int, int], blocker: Tuple[int, int], 
343                          target: Tuple[int, int]) -> bool:
344        """Check if target is behind blocker from viewer's perspective."""
345        # Simple check: if blocker is between viewer and target on the line
346        hex_grid = HexGrid()
347        viewer_hex = hex_grid.offset_to_axial(viewer[0], viewer[1])
348        blocker_hex = hex_grid.offset_to_axial(blocker[0], blocker[1])
349        target_hex = hex_grid.offset_to_axial(target[0], target[1])
350        
351        # Calculate distances
352        viewer_to_blocker = viewer_hex.distance_to(blocker_hex)
353        viewer_to_target = viewer_hex.distance_to(target_hex)
354        
355        # Target is behind if it's further than blocker
356        return viewer_to_target > viewer_to_blocker

362    @staticmethod
363    def _get_line(start: HexCoord, end: HexCoord) -> List[HexCoord]:
364        """Get a line of hexes between two points"""
365        distance = start.distance_to(end)
366        results = []
367        
368        if distance == 0:
369            return [start]
370            
371        # Linear interpolation in cube coordinates
372        for i in range(distance + 1):
373            t = i / distance
374            
375            # Convert to cube coordinates
376            start_cube = start.to_cube()
377            end_cube = end.to_cube()
378            
379            # Interpolate
380            x = start_cube[0] + (end_cube[0] - start_cube[0]) * t
381            y = start_cube[1] + (end_cube[1] - start_cube[1]) * t
382            z = start_cube[2] + (end_cube[2] - start_cube[2]) * t
383            
384            # Round to nearest hex
385            rx = round(x)
386            ry = round(y)
387            rz = round(z)
388            
389            # Fix rounding errors
390            x_diff = abs(rx - x)
391            y_diff = abs(ry - y)
392            z_diff = abs(rz - z)
393            
394            if x_diff > y_diff and x_diff > z_diff:
395                rx = -ry - rz
396            elif y_diff > z_diff:
397                ry = -rx - rz
398            else:
399                rz = -rx - ry
400                
401            # Convert back to axial
402            results.append(HexCoord(rx, rz))
403            
404        return results

458    def update_player_visibility_incremental(self, game_state, player_id: int, 
459                                           changed_positions: Optional[List[Tuple[int, int]]] = None):
460        """
461        Incrementally update visibility for positions that might have changed.
462        More efficient than full recalculation.
463        
464        Args:
465            game_state: Current game state
466            player_id: Player to update visibility for
467            changed_positions: List of positions where units moved to/from
468        """
469        if changed_positions is None:
470            # Fall back to full update
471            self.update_player_visibility(game_state, player_id)
472            return
473            
474        # Cache game_state
475        self._cached_game_state = game_state
476        
477        # For each changed position, update visibility in that area
478        updated_hexes = set()
479        max_vision_range = 5  # Maximum possible vision range
480        
481        for pos in changed_positions:
482            # Get hexes that might be affected by change at this position
483            hex_grid = HexGrid()
484            center_hex = hex_grid.offset_to_axial(pos[0], pos[1])
485            
486            # Check all hexes within max vision range
487            for y in range(max(0, pos[1] - max_vision_range), 
488                          min(self.height, pos[1] + max_vision_range + 1)):
489                for x in range(max(0, pos[0] - max_vision_range), 
490                              min(self.width, pos[0] + max_vision_range + 1)):
491                    target_hex = hex_grid.offset_to_axial(x, y)
492                    if center_hex.distance_to(target_hex) <= max_vision_range:
493                        updated_hexes.add((x, y))
494                        
495        # Now recalculate visibility only for affected hexes
496        # First downgrade existing visibility in affected areas
497        for hex_pos in updated_hexes:
498            if hex_pos in self.visibility_maps[player_id]:
499                current = self.visibility_maps[player_id][hex_pos]
500                if current in [VisibilityState.VISIBLE, VisibilityState.PARTIAL]:
501                    self.visibility_maps[player_id][hex_pos] = VisibilityState.EXPLORED
502                    
503        # Then calculate new visibility from player's units
504        for unit in game_state.units:
505            if unit.player_id == player_id:
506                vision_range = self._get_unit_vision_range(unit)
507                vision_behavior = unit.get_behavior('VisionBehavior') if hasattr(unit, 'get_behavior') else None
508                is_elevated = vision_behavior.is_elevated() if vision_behavior else False
509                
510                # Only check visibility for affected hexes within this unit's range
511                for hex_pos in updated_hexes:
512                    hex_grid = HexGrid()
513                    unit_hex = hex_grid.offset_to_axial(unit.x, unit.y)
514                    target_hex = hex_grid.offset_to_axial(hex_pos[0], hex_pos[1])
515                    distance = unit_hex.distance_to(target_hex)
516                    
517                    if distance <= vision_range:
518                        if self._has_line_of_sight(game_state, (unit.x, unit.y), hex_pos, is_elevated):
519                            # Update visibility state based on distance
520                            if distance <= self.vision_config.full_id_range:
521                                new_state = VisibilityState.VISIBLE
522                            elif distance <= self.vision_config.partial_id_range:
523                                new_state = VisibilityState.PARTIAL
524                            else:
525                                new_state = VisibilityState.EXPLORED
526                                
527                            current_state = self.visibility_maps[player_id].get(hex_pos, VisibilityState.HIDDEN)
528                            if new_state.value > current_state.value:
529                                self.visibility_maps[player_id][hex_pos] = new_state

268    def _has_line_of_sight(self, game_state, origin: Tuple[int, int], 
269                          target: Tuple[int, int], is_elevated: bool) -> bool:
270        """Check if there's line of sight between two hexes."""
271        # Simple line of sight - check each hex along the path
272        hex_grid = HexGrid()
273        origin_hex = hex_grid.offset_to_axial(origin[0], origin[1])
274        target_hex = hex_grid.offset_to_axial(target[0], target[1])
275        
276        # Get line between hexes
277        line_hexes = self._get_line(origin_hex, target_hex)
278        
279        # Check each hex along the line for blocking
280        for i, hex_coord in enumerate(line_hexes[1:-1], 1):  # Skip origin and target
281            # Convert back to offset coordinates
282            hex_x, hex_y = hex_grid.axial_to_offset(hex_coord)
283            
284            # Check terrain blocking
285            terrain = game_state.terrain_map.get_terrain(hex_x, hex_y)
286            
287            if terrain:
288                terrain_type = terrain.type.value.lower()
289                
290                # Mountains always block line of sight (impassable high terrain)
291                if terrain_type == 'mountains':
292                    return False
293                
294                # High hills block unless viewer is elevated or on high ground
295                if terrain_type == 'high hills':
296                    origin_terrain = game_state.terrain_map.get_terrain(origin[0], origin[1])
297                    viewer_on_high_ground = origin_terrain and origin_terrain.type.value.lower() in ['hills', 'high hills', 'mountains']
298                    if not is_elevated and not viewer_on_high_ground:
299                        return False
300                
301                # Hills block unless viewer is elevated or on a hill
302                if terrain_type == 'hills':
303                    origin_terrain = game_state.terrain_map.get_terrain(origin[0], origin[1])
304                    # If viewer is on hills or is elevated (cavalry), they can see over hills
305                    viewer_on_hills = origin_terrain and origin_terrain.type.value.lower() in ['hills', 'high hills', 'mountains']
306                    if not is_elevated and not viewer_on_hills:
307                        return False
308            
309            # Check castle blocking
310            if hasattr(game_state, 'castles'):
311                for castle in game_state.castles:
312                    if hasattr(castle, 'contains_position') and castle.contains_position(hex_x, hex_y):
313                        # Castles are tall, but unless viewer is elevated or on a mountain, they block LOS
314                        origin_terrain = game_state.terrain_map.get_terrain(origin[0], origin[1])
315                        viewer_on_mountain = origin_terrain and origin_terrain.type.value.lower() == 'mountains'
316                        if not is_elevated and not viewer_on_mountain:
317                            return False
318                    
319            # Check unit blocking
320            blocking_unit = game_state.get_unit_at(hex_x, hex_y)
321            if blocking_unit:
322                # Check if unit blocks vision using its behavior
323                vision_behavior = blocking_unit.get_behavior('VisionBehavior') if hasattr(blocking_unit, 'get_behavior') else None
324                if vision_behavior and vision_behavior.blocks_vision():
325                    # Check if blocker is also elevated
326                    blocker_elevated = vision_behavior.is_elevated() if hasattr(vision_behavior, 'is_elevated') else False
327                    
328                    # Elevated viewers can see over non-elevated blockers
329                    if is_elevated and not blocker_elevated:
330                        continue
331                        
332                    # If both are elevated (e.g., cavalry viewing past cavalry), no blocking
333                    if is_elevated and blocker_elevated:
334                        continue
335                        
336                    # Otherwise check if target is behind the blocker
337                    if self._is_behind_blocker(origin, (hex_x, hex_y), target):
338                        return False
339                        
340        return True

363    @staticmethod
364    def _get_line(start: HexCoord, end: HexCoord) -> List[HexCoord]:
365        """Get a line of hexes between two points"""
366        distance = start.distance_to(end)
367        results = []
368        
369        if distance == 0:
370            return [start]
371            
372        # Linear interpolation in cube coordinates
373        for i in range(distance + 1):
374            t = i / distance
375            
376            # Convert to cube coordinates
377            start_cube = start.to_cube()
378            end_cube = end.to_cube()
379            
380            # Interpolate
381            x = start_cube[0] + (end_cube[0] - start_cube[0]) * t
382            y = start_cube[1] + (end_cube[1] - start_cube[1]) * t
383            z = start_cube[2] + (end_cube[2] - start_cube[2]) * t
384            
385            # Round to nearest hex
386            rx = round(x)
387            ry = round(y)
388            rz = round(z)
389            
390            # Fix rounding errors
391            x_diff = abs(rx - x)
392            y_diff = abs(ry - y)
393            z_diff = abs(rz - z)
394            
395            if x_diff > y_diff and x_diff > z_diff:
396                rx = -ry - rz
397            elif y_diff > z_diff:
398                ry = -rx - rz
399            else:
400                rz = -rx - ry
401                
402            # Convert back to axial
403            results.append(HexCoord(rx, rz))
404            
405        return results

459    def update_player_visibility_incremental(self, game_state, player_id: int, 
460                                           changed_positions: Optional[List[Tuple[int, int]]] = None):
461        """
462        Incrementally update visibility for positions that might have changed.
463        More efficient than full recalculation.
464        
465        Args:
466            game_state: Current game state
467            player_id: Player to update visibility for
468            changed_positions: List of positions where units moved to/from
469        """
470        if changed_positions is None:
471            # Fall back to full update
472            self.update_player_visibility(game_state, player_id)
473            return
474            
475        # Cache game_state
476        self._cached_game_state = game_state
477        
478        # For each changed position, update visibility in that area
479        updated_hexes = set()
480        max_vision_range = 5  # Maximum possible vision range
481        
482        for pos in changed_positions:
483            # Get hexes that might be affected by change at this position
484            hex_grid = HexGrid()
485            center_hex = hex_grid.offset_to_axial(pos[0], pos[1])
486            
487            # Check all hexes within max vision range
488            for y in range(max(0, pos[1] - max_vision_range), 
489                          min(self.height, pos[1] + max_vision_range + 1)):
490                for x in range(max(0, pos[0] - max_vision_range), 
491                              min(self.width, pos[0] + max_vision_range + 1)):
492                    target_hex = hex_grid.offset_to_axial(x, y)
493                    if center_hex.distance_to(target_hex) <= max_vision_range:
494                        updated_hexes.add((x, y))
495                        
496        # Now recalculate visibility only for affected hexes
497        # First downgrade existing visibility in affected areas
498        for hex_pos in updated_hexes:
499            if hex_pos in self.visibility_maps[player_id]:
500                current = self.visibility_maps[player_id][hex_pos]
501                if current in [VisibilityState.VISIBLE, VisibilityState.PARTIAL]:
502                    self.visibility_maps[player_id][hex_pos] = VisibilityState.EXPLORED
503                    
504        # Then calculate new visibility from player's units
505        for unit in game_state.units:
506            if unit.player_id == player_id:
507                vision_range = self._get_unit_vision_range(unit)
508                vision_behavior = unit.get_behavior('VisionBehavior') if hasattr(unit, 'get_behavior') else None
509                is_elevated = vision_behavior.is_elevated() if vision_behavior else False
510                
511                # Only check visibility for affected hexes within this unit's range
512                for hex_pos in updated_hexes:
513                    hex_grid = HexGrid()
514                    unit_hex = hex_grid.offset_to_axial(unit.x, unit.y)
515                    target_hex = hex_grid.offset_to_axial(hex_pos[0], hex_pos[1])
516                    distance = unit_hex.distance_to(target_hex)
517                    
518                    if distance <= vision_range:
519                        if self._has_line_of_sight(game_state, (unit.x, unit.y), hex_pos, is_elevated):
520                            # Update visibility state based on distance
521                            if distance <= self.vision_config.full_id_range:
522                                new_state = VisibilityState.VISIBLE
523                            elif distance <= self.vision_config.partial_id_range:
524                                new_state = VisibilityState.PARTIAL

527                                
528                            current_state = self.visibility_maps[player_id].get(hex_pos, VisibilityState.HIDDEN)
529                            if new_state.value > current_state.value:
530                                self.visibility_maps[player_id][hex_pos] = new_state
