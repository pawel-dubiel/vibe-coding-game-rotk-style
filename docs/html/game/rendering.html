 49    def render_terrain(self, game_state):
 50        """Render the hex grid terrain with fog of war."""
 51        # Calculate visible hex range using camera manager for zoom-aware coordinates
 52        if not hasattr(game_state, 'camera_manager'):
 53            raise ValueError("game_state.camera_manager is required for terrain rendering")
 54
 55        camera_x = game_state.camera_manager.camera_x
 56        camera_y = game_state.camera_manager.camera_y
 57        screen_width = game_state.camera_manager.screen_width
 58        screen_height = game_state.camera_manager.screen_height
 59
 60        # Calculate visible hex range
 61        # The hex_layout dimensions are already scaled by zoom, and the camera coordinates
 62        # are in the same scaled space, so we use them directly
 63        hex_width = self.hex_layout.col_spacing
 64        hex_height = self.hex_layout.row_spacing
 65
 66        start_col = max(0, int(camera_x / hex_width) - 1)
 67        end_col = min(game_state.board_width, int((camera_x + screen_width) / hex_width) + 2)
 68        start_row = max(0, int(camera_y / hex_height) - 1)
 69        end_row = min(game_state.board_height, int((camera_y + screen_height) / hex_height) + 2)
 70        
 71        for col in range(start_col, end_col):
 72            for row in range(start_row, end_row):
 73                self._render_hex_tile(game_state, col, row)
 74    
 75    def _render_hex_tile(self, game_state, col: int, row: int):
 76        """Render a single hex tile with terrain and fog."""
 77        # Calculate pixel position
 78        pixel_x, pixel_y = self.hex_layout.hex_to_pixel(col, row)
 79        screen_x, screen_y = game_state.world_to_screen(pixel_x, pixel_y)
 80        
 81        # Get terrain and color
 82        terrain = game_state.terrain_map.get_terrain(col, row)
 83        color = self._get_terrain_color(terrain, col, row)
 84        
 85        # Get hex corners
 86        corners = self.hex_layout.get_hex_corners(screen_x, screen_y)
 87        
 88        # Try to use asset image, fallback to colored polygon
 89        terrain_image = self._get_terrain_asset(terrain)
 90        
 91        if terrain_image:
 92            image_rect = terrain_image.get_rect(center=(int(screen_x), int(screen_y)))
 93            self.screen.blit(terrain_image, image_rect)
 94            pygame.draw.polygon(self.screen, (0, 0, 0), corners, 1)  # Black outline
 95        else:
 96            # Fallback to colored polygon
 97            pygame.draw.polygon(self.screen, color, corners)
 98            pygame.draw.polygon(self.screen, (0, 0, 0), corners, 1)  # Black outline
 99            
100            # Add texture patterns for terrains without assets
101            self._add_terrain_texture(terrain, screen_x, screen_y, corners)
102        
103        # Apply fog of war overlay
104        self._apply_fog_overlay(game_state, col, row, screen_x, screen_y, corners)
105        
106        # Draw coordinate labels if enabled
107        if getattr(game_state, 'show_coordinates', False):
108            self._draw_coordinates(col, row, screen_x, screen_y)

110    def _get_terrain_color(self, terrain, col: int, row: int) -> tuple:
111        """Get the base color for a terrain tile."""
112        if terrain:
113            return self.terrain_colors.get(terrain.type, self.terrain_colors[TerrainType.PLAINS])
114        else:
115            # Alternating pattern for default tiles
116            return self.tile_light if (col + row) % 2 == 0 else self.tile_dark

118    def _get_terrain_asset(self, terrain):
119        """Get the asset image for a terrain type."""
120        if not terrain:
121            return None
122        
123        terrain_asset_map = {
124            TerrainType.WATER: "water",
125            TerrainType.DEEP_WATER: "deep-water",
126            TerrainType.PLAINS: "plain",
127            TerrainType.HILLS: "hills",
128            TerrainType.LIGHT_FOREST: "light-forrest",
129            TerrainType.FOREST: "forrest",
130            TerrainType.DENSE_FOREST: "dense-forrest",
131            TerrainType.SWAMP: "swamp",
132            TerrainType.MOUNTAINS: "mountains",
133            TerrainType.HIGH_HILLS: "high-hills",
134            TerrainType.SNOW: "snow",
135            TerrainType.DESERT: "desert"
136        }
137        
138        asset_name = terrain_asset_map.get(terrain.type)
139        if asset_name:
140            return self.asset_manager.get_terrain_image(asset_name, self.hex_grid.hex_size)
141        
142        return None

144    def _add_terrain_texture(self, terrain, screen_x: float, screen_y: float, corners: list):
145        """Add texture patterns for terrains without asset images."""
146        if not terrain:
147            return
148        
149        # Add small visual elements to distinguish terrain types
150        if terrain.type == TerrainType.FOREST:
151            # Draw small trees
152            for i in range(3):
153                tree_x = screen_x + (i - 1) * 10
154                tree_y = screen_y + (i % 2) * 8
155                pygame.draw.circle(self.screen, (40, 80, 40), (int(tree_x), int(tree_y)), 3)
156        
157        elif terrain.type == TerrainType.HILLS:
158            # Draw hill contours
159            for i in range(2):
160                hill_radius = 15 + i * 5
161                pygame.draw.circle(self.screen, (160, 140, 110), 
162                                 (int(screen_x), int(screen_y)), hill_radius, 1)
163        
164        elif terrain.type == TerrainType.WATER:
165            # Draw wave patterns
166            for i in range(3):
167                wave_y = screen_y + (i - 1) * 8
168                pygame.draw.arc(self.screen, (100, 140, 220),
169                              (screen_x - 15, wave_y - 3, 30, 6), 0, math.pi, 1)
170        
171        elif terrain.type == TerrainType.MOUNTAINS:
172            # Draw mountain peaks
173            peak_points = [
174                (screen_x - 10, screen_y + 8),
175                (screen_x - 5, screen_y - 8),
176                (screen_x, screen_y + 5),
177                (screen_x + 5, screen_y - 8),
178                (screen_x + 10, screen_y + 8)
179            ]
180            pygame.draw.lines(self.screen, (100, 80, 60), False, peak_points, 2)

182    def _apply_fog_overlay(self, game_state, col: int, row: int, screen_x: float, screen_y: float, corners: list):
183        """Apply fog of war overlay based on visibility state."""
184        # Check fog of war visibility
185        visibility = VisibilityState.VISIBLE  # Default for no fog
186        if hasattr(game_state, 'fog_of_war') and game_state.fog_view_player is not None:
187            visibility = game_state.fog_of_war.get_visibility_state(game_state.fog_view_player, col, row)
188        
189        # Apply fog overlay for non-visible hexes
190        if visibility != VisibilityState.VISIBLE:
191            fog_surface = pygame.Surface((int(self.hex_grid.hex_width), int(self.hex_grid.hex_height)), pygame.SRCALPHA)
192            
193            # Different alpha values for different visibility states
194            if visibility == VisibilityState.HIDDEN:
195                fog_color = (40, 40, 40, 200)  # Dark grey, very opaque
196            elif visibility == VisibilityState.EXPLORED:
197                fog_color = (60, 60, 60, 150)  # Grey, semi-opaque
198            else:  # PARTIAL
199                fog_color = (80, 80, 80, 100)  # Light grey, less opaque
200            
201            # Draw fog polygon on the surface
202            corners_relative = [(x - screen_x + self.hex_grid.hex_width/2, 
203                               y - screen_y + self.hex_grid.hex_height/2) for x, y in corners]
204            pygame.draw.polygon(fog_surface, fog_color, corners_relative)
205            
206            # Blit the fog overlay
207            self.screen.blit(fog_surface, (screen_x - self.hex_grid.hex_width/2, 
208                                          screen_y - self.hex_grid.hex_height/2))

201        """Draw a dotted line between two points."""
202        x1, y1 = start_pos
203        x2, y2 = end_pos

205        # Calculate total distance and direction
206        dx = x2 - x1
207        dy = y2 - y1
208        distance = math.sqrt(dx * dx + dy * dy)
209

212

214        dx /= distance
215        dy /= distance
216        
217        # Draw dashes

220            # Start of dash
221            start_x = x1 + dx * current_distance
222            start_y = y1 + dy * current_distance
223            
224            # End of dash
225            end_distance = min(current_distance + dash_length, distance)
226            end_x = x1 + dx * end_distance
227            end_y = y1 + dy * end_distance
228            
229            # Draw dash
230            pygame.draw.line(surface, color, (start_x, start_y), (end_x, end_y), width)
231            
232            # Move to next dash (skip gap)
233            current_distance += dash_length * 2
