 11class Shadow:
 12    """Represents a shadow/blocked area in the field of view"""
 13    start: float  # Start angle of shadow
 14    end: float    # End angle of shadow
 15    
 16    def contains(self, angle: float) -> bool:
 17        """Check if an angle is within this shadow"""
 18        return self.start <= angle <= self.end

 21        """Check if this shadow completely blocks an arc"""
 22        return self.start <= start and self.end >= end

 25class HexShadowcaster:
 26    """
 27    Efficient shadow casting for hexagonal grids.
 28    Calculates visible hexes from a point using shadow propagation.
 29    """
 30    
 31    def __init__(self):
 32        self.hex_grid = HexGrid()
 33        # Pre-calculate hex directions for the 6 sextants
 34        self._init_sextants()
 35        
 36    def _init_sextants(self):
 37        """Initialize the 6 sextants (60-degree wedges) for hex grids"""
 38        # Each sextant covers 60 degrees of the hex grid
 39        # We'll process each sextant separately to handle symmetry
 40        self.sextants = []
 41        
 42        # Define the 6 directions and their angular ranges
 43        # In a hex grid, we have natural 60-degree divisions
 44        for i in range(6):
 45            start_angle = i * 60.0
 46            end_angle = (i + 1) * 60.0
 47            self.sextants.append({
 48                'id': i,
 49                'start_angle': start_angle,
 50                'end_angle': end_angle,
 51                'direction': i  # Maps to hex directions
 52            })
 53    
 54    def calculate_visible_hexes(self, game_state, origin: Tuple[int, int], 
 55                               max_range: int, viewer_elevation: float = 0.0) -> Dict[Tuple[int, int], int]:
 56        """
 57        Calculate all visible hexes from origin using shadow casting.
 58        
 59        Args:
 60            game_state: Game state for terrain/unit queries
 61            origin: Starting position (x, y) in offset coordinates
 62            max_range: Maximum vision range
 63            viewer_elevation: Height of viewer (0 for ground, positive for elevated)
 64            
 65        Returns:
 66            Dict mapping visible positions to their distance from origin
 67        """
 68        visible_hexes = {origin: 0}  # Always see your own position
 69        
 70        # Convert origin to hex coordinates
 71        origin_hex = self.hex_grid.offset_to_axial(origin[0], origin[1])
 72        
 73        # Process each of the 6 sextants
 74        for sextant in self.sextants:
 75            self._cast_shadows_in_sextant(
 76                game_state, origin, origin_hex, max_range, 
 77                viewer_elevation, sextant['direction'], visible_hexes
 78            )
 79            
 80        return visible_hexes

 82    def _cast_shadows_in_sextant(self, game_state, origin: Tuple[int, int],
 83                                 origin_hex: HexCoord, max_range: int,
 84                                 viewer_elevation: float, direction: int,
 85                                 visible_hexes: Dict[Tuple[int, int], int]):
 86        """
 87        Cast shadows in one 60-degree sextant using recursive shadow casting.
 88        """
 89        # Track shadows as we go outward
 90        shadows: List[Shadow] = []
 91        
 92        # Process each ring of hexes outward from origin
 93        for distance in range(1, max_range + 1):
 94            # Get hexes at this distance in this sextant
 95            ring_hexes = self._get_ring_hexes_in_sextant(origin_hex, distance, direction)
 96            
 97            for hex_pos, angle_start, angle_end in ring_hexes:
 98                # Convert back to offset coordinates
 99                offset_x, offset_y = self.hex_grid.axial_to_offset(hex_pos)
100                
101                # Check if this hex is in valid bounds
102                if not (0 <= offset_x < game_state.board_width and 
103                       0 <= offset_y < game_state.board_height):
104                    continue
105                
106                # Check if this hex is completely shadowed
107                if self._is_shadowed(angle_start, angle_end, shadows):
108                    continue
109                    
110                # This hex is at least partially visible
111                offset_pos = (offset_x, offset_y)
112                visible_hexes[offset_pos] = distance
113                
114                # Check if this hex blocks vision
115                if self._blocks_vision(game_state, origin, offset_pos, viewer_elevation):
116                    # Add shadow for this blocking hex
117                    self._add_shadow(angle_start, angle_end, shadows)
118    
119    def _get_ring_hexes_in_sextant(self, center: HexCoord, radius: int, 
120                                   direction: int) -> List[Tuple[HexCoord, float, float]]:
121        """
122        Get all hexes at given radius in a specific sextant (60-degree wedge).
123        Returns list of (hex_coord, start_angle, end_angle) tuples.
124        """
125        results = []
126        
127        # Use hex ring algorithm but filter to sextant
128        # Start from the direction and go 1/6 of the way around
129        directions = [
130            HexCoord(1, 0), HexCoord(1, -1), HexCoord(0, -1),
131            HexCoord(-1, 0), HexCoord(-1, 1), HexCoord(0, 1)
132        ]
133        
134        # Start position for this ring
135        current = center
136        for _ in range(radius):
137            current = HexCoord(current.q + directions[direction].q,
138                             current.r + directions[direction].r)
139        
140        # Walk 1/6 of the ring (for this sextant)
141        steps_per_sextant = radius
142        dir_index = (direction + 2) % 6  # Perpendicular direction
143        
144        for step in range(steps_per_sextant):
145            # Calculate angular position of this hex
146            angle = self._calculate_hex_angle(center, current)
147            # Hex "width" in angular terms depends on distance
148            angular_width = 60.0 / max(1, radius)  # Approximation
149            
150            results.append((
151                current,
152                angle - angular_width / 2,
153                angle + angular_width / 2
154            ))
155            
156            # Move to next hex in ring
157            current = HexCoord(current.q + directions[dir_index].q,
158                             current.r + directions[dir_index].r)
159            
160        return results
161    
162    def _calculate_hex_angle(self, origin: HexCoord, target: HexCoord) -> float:
163        """Calculate angle from origin to target hex (in degrees)"""
164        # Convert to cube coordinates for easier calculation
165        ox, oy, oz = origin.to_cube()
166        tx, ty, tz = target.to_cube()
167        
168        # Calculate angle using cube coordinate differences
169        dx = tx - ox
170        dy = ty - oy
171        
172        # Convert to angle (0-360 degrees)
173        # This is simplified for hex grids
174        if dx == 0 and dy == 0:
175            return 0.0
176            
177        # Use the hex direction to determine angle
178        # Each of the 6 directions is 60 degrees apart
179        if dx > 0 and dy <= 0:
180            base_angle = 0.0
181        elif dx == 0 and dy < 0:
182            base_angle = 60.0
183        elif dx < 0 and dy < 0:
184            base_angle = 120.0
185        elif dx < 0 and dy >= 0:
186            base_angle = 180.0
187        elif dx == 0 and dy > 0:
188            base_angle = 240.0
189        else:  # dx > 0 and dy > 0
190            base_angle = 300.0
191            
192        # Add fractional angle based on position within sextant
193        return base_angle

195    def _is_shadowed(self, start_angle: float, end_angle: float, 
196                     shadows: List[Shadow]) -> bool:
197        """Check if an angular range is completely in shadow"""
198        for shadow in shadows:
199            if shadow.blocks(start_angle, end_angle):
200                return True
201        return False

203    def _blocks_vision(self, game_state, origin: Tuple[int, int],
204                      target: Tuple[int, int], viewer_elevation: float) -> bool:
205        """
206        Check if a hex blocks vision based on terrain and units.
207        """
208        # Get terrain at target
209        terrain = game_state.terrain_map.get_terrain(target[0], target[1])
210        
211        # Hills block unless viewer is elevated
212        if terrain and terrain.type.value.lower() == 'hills':
213            origin_terrain = game_state.terrain_map.get_terrain(origin[0], origin[1])
214            viewer_on_hills = origin_terrain and origin_terrain.type.value.lower() == 'hills'
215            
216            if viewer_elevation <= 0 and not viewer_on_hills:
217                return True  # Hills block vision
218                
219        # Mountains always block
220        if terrain and terrain.type.value.lower() == 'mountains':
221            return True
222            
223        # Check for castle blocking
224        if hasattr(game_state, 'castles'):
225            for castle in game_state.castles:
226                if hasattr(castle, 'contains_position') and castle.contains_position(target[0], target[1]):
227                    # Castles block vision unless viewer is elevated or on a mountain
228                    origin_terrain = game_state.terrain_map.get_terrain(origin[0], origin[1])
229                    viewer_on_mountain = origin_terrain and origin_terrain.type.value.lower() == 'mountains'
230                    if viewer_elevation <= 0 and not viewer_on_mountain:
231                        return True
232            
233        # Check for blocking units
234        unit = game_state.get_unit_at(target[0], target[1])
235        if unit and unit.player_id != game_state.current_player:
236            # Enemy units can block vision
237            vision_behavior = unit.get_behavior('VisionBehavior') if hasattr(unit, 'get_behavior') else None
238            if vision_behavior and vision_behavior.blocks_vision():
239                # Check elevation interaction
240                unit_elevation = 1.0 if vision_behavior.is_elevated() else 0.0
241                
242                # Higher elevation can see over lower blockers
243                if viewer_elevation > unit_elevation:
244                    return False
245                    
246                return True
247                
248        return False
249    
250    def _add_shadow(self, start_angle: float, end_angle: float, 
251                   shadows: List[Shadow]):
252        """Add a shadow to the list, merging with existing shadows if needed"""
253        new_shadow = Shadow(start_angle, end_angle)
254        
255        # Simple implementation: just add the shadow
256        # A more sophisticated version would merge overlapping shadows
257        shadows.append(new_shadow)
258        
259        # Sort shadows by start angle for easier processing
260        shadows.sort(key=lambda s: s.start)
261        
262        # Merge overlapping shadows
263        merged = []
264        for shadow in shadows:
265            if not merged:
266                merged.append(shadow)
267            else:
268                last = merged[-1]
269                if shadow.start <= last.end:
270                    # Overlapping, merge them
271                    last.end = max(last.end, shadow.end)
272                else:
273                    merged.append(shadow)
274                    
275        shadows[:] = merged

278class SimpleShadowcaster:
279    """
280    Simpler shadow casting implementation that's more suitable for hex grids.
281    Uses a sector-based approach rather than true shadow casting.
282    """
283    
284    def __init__(self):
285        self.hex_grid = HexGrid()
286    
287    def calculate_visible_hexes(self, game_state, origin: Tuple[int, int],
288                               max_range: int, is_elevated: bool = False) -> Dict[Tuple[int, int], int]:
289        """
290        Calculate visible hexes using a simplified shadow casting approach.
291        More efficient than checking line-of-sight to every hex.
292        """
293        visible_hexes = {origin: 0}
294        origin_hex = self.hex_grid.offset_to_axial(origin[0], origin[1])
295        
296        # Instead of tracking blocked sectors, check line of sight for each hex
297        # This is more accurate for hex grids where "sectors" don't map cleanly
298        
299        # Process each ring of hexes
300        for distance in range(1, max_range + 1):
301            # Get all hexes at this distance
302            ring_hexes = SimpleShadowcaster._get_hex_ring(origin_hex, distance)
303            
304            for hex_coord in ring_hexes:
305                # Convert to offset coordinates
306                offset_x, offset_y = self.hex_grid.axial_to_offset(hex_coord)
307                offset_pos = (offset_x, offset_y)
308                
309                # Check bounds
310                if not (0 <= offset_x < game_state.board_width and 
311                       0 <= offset_y < game_state.board_height):
312                    continue
313                
314                # Check if we can see this hex using line of sight
315                if SimpleShadowcaster._check_visibility_simple(game_state, origin, offset_pos, is_elevated):
316                    visible_hexes[offset_pos] = distance
317                        
318        return visible_hexes

320    @staticmethod
321    def _get_hex_ring(center: HexCoord, radius: int) -> List[HexCoord]:
322        """Get all hexes at exactly 'radius' distance from center"""
323        if radius == 0:
324            return [center]
325            
326        results = []
327        directions = [
328            HexCoord(1, 0), HexCoord(1, -1), HexCoord(0, -1),
329            HexCoord(-1, 0), HexCoord(-1, 1), HexCoord(0, 1)
330        ]
331        
332        # Start at a corner of the ring
333        current = center
334        for _ in range(radius):
335            current = HexCoord(current.q + directions[4].q, current.r + directions[4].r)
336            
337        # Walk around the ring
338        for i in range(6):
339            for _ in range(radius):
340                results.append(current)
341                current = HexCoord(current.q + directions[i].q, current.r + directions[i].r)
342                
343        return results

345    @staticmethod
346    def _get_sector(origin: HexCoord, target: HexCoord) -> int:
347        """
348        Determine which of the 6 hex sectors the target is in relative to origin.
349        Returns 0-5 for the 6 directions.
350        """
351        dx = target.q - origin.q
352        dy = target.r - origin.r
353        
354        # Determine primary direction
355        if dx > 0 and dy <= 0:
356            return 0  # East
357        elif dx == 0 and dy < 0:
358            return 1  # Northeast  
359        elif dx < 0 and dy < 0:
360            return 2  # Northwest
361        elif dx < 0 and dy >= 0:
362            return 3  # West
363        elif dx == 0 and dy > 0:
364            return 4  # Southwest
365        else:
366            return 5  # Southeast

368    @staticmethod
369    def _check_visibility_simple(game_state, origin: Tuple[int, int],
370                                target: Tuple[int, int], is_elevated: bool) -> bool:
371        """Check visibility along the line between origin and target"""
372        hex_grid = HexGrid()
373        origin_hex = hex_grid.offset_to_axial(origin[0], origin[1])
374        target_hex = hex_grid.offset_to_axial(target[0], target[1])
375        
376        # Get all hexes along the line
377        line_hexes = HexGrid.get_line(origin_hex, target_hex)
378        
379        # Check each hex for blocking (except origin and target)
380        for i, hex_coord in enumerate(line_hexes[1:-1], 1):
381            offset_x, offset_y = hex_grid.axial_to_offset(hex_coord)
382            
383            # Check bounds
384            if not (0 <= offset_x < game_state.board_width and 
385                   0 <= offset_y < game_state.board_height):
386                continue
387                
388            # Check if this hex blocks vision to the target
389            # Note: we're checking if the hex at (offset_x, offset_y) blocks vision from origin
390            if SimpleShadowcaster._blocks_vision_simple(game_state, origin, (offset_x, offset_y), is_elevated):
391                return False
392                
393        return True
394        
395    @staticmethod
396    def _blocks_vision_simple(game_state, origin: Tuple[int, int],
397                             target: Tuple[int, int], is_elevated: bool) -> bool:
398        """Check if this hex blocks further vision"""
399        # Don't block at origin
400        if target == origin:
401            return False
402            
403        terrain = game_state.terrain_map.get_terrain(target[0], target[1])
404        
405        # Mountains always block
406        if terrain and terrain.type.value.lower() == 'mountains':
407            return True
408            
409        # Hills block unless viewer is elevated
410        if terrain and terrain.type.value.lower() == 'hills':
411            origin_terrain = game_state.terrain_map.get_terrain(origin[0], origin[1])
412            viewer_on_hills = origin_terrain and origin_terrain.type.value.lower() == 'hills'
413            
414            if not is_elevated and not viewer_on_hills:
415                return True
416        
417        # Check for castle blocking
418        if hasattr(game_state, 'castles'):
419            for castle in game_state.castles:
420                if hasattr(castle, 'contains_position') and castle.contains_position(target[0], target[1]):
421                    # Castles block vision unless viewer is elevated or on a mountain
422                    origin_terrain = game_state.terrain_map.get_terrain(origin[0], origin[1])
423                    viewer_on_mountain = origin_terrain and origin_terrain.type.value.lower() == 'mountains'
424                    if not is_elevated and not viewer_on_mountain:
425                        # print(f"DEBUG: Castle at {target} blocks vision from {origin}")
426                        return True
427        # Check units
428        unit = game_state.get_unit_at(target[0], target[1])
429        if unit:
430            vision_behavior = unit.get_behavior('VisionBehavior') if hasattr(unit, 'get_behavior') else None
431            if vision_behavior and vision_behavior.blocks_vision():
432                # Elevated viewers can see over non-elevated blockers
433                if is_elevated and not vision_behavior.is_elevated():
434                    return False
435                return True
436                
437        return False

233            
234        # Check for blocking units

236        if unit and unit.player_id != game_state.current_player:

239            if vision_behavior and vision_behavior.blocks_vision():

244                if viewer_elevation > unit_elevation:

274                    merged.append(shadow)

323        """Get all hexes at exactly 'radius' distance from center"""
324        if radius == 0:
325            return [center]
326            
327        results = []
328        directions = [

330            HexCoord(-1, 0), HexCoord(-1, 1), HexCoord(0, 1)
331        ]
332        
333        # Start at a corner of the ring
334        current = center
335        for _ in range(radius):
336            current = HexCoord(current.q + directions[4].q, current.r + directions[4].r)
337            
338        # Walk around the ring
339        for i in range(6):
340            for _ in range(radius):
341                results.append(current)

348        """

350        Returns 0-5 for the 6 directions.

355        # Determine primary direction
356        if dx > 0 and dy <= 0:

358        elif dx == 0 and dy < 0:

361            return 2  # Northwest
