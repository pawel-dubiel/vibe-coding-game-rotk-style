 33        self.save_dir = self._get_save_directory()

 37    def _get_save_directory(self) -> str:
 38        """Get the absolute path to the save directory"""
 39        # Get the directory where this script is located
 40        current_dir = os.path.dirname(os.path.abspath(__file__))
 41        # Go up one level to game root
 42        game_root = os.path.dirname(current_dir)
 43        # Create saves directory path
 44        return os.path.join(game_root, self.SAVE_DIR)
 45        
 46    def _ensure_save_directory(self):
 47        """Create save directory if it doesn't exist"""
 48        os.makedirs(self.save_dir, exist_ok=True)
 49        
 50    def _get_save_filename(self, slot: int) -> str:
 51        """Get filename for a save slot"""
 52        return os.path.join(self.save_dir, f"save_slot_{slot}.pkl")
 53        
 54    def _get_metadata_path(self) -> str:
 55        """Get path to metadata file"""
 56        return os.path.join(self.save_dir, self.METADATA_FILE)
 57        
 58    def _load_metadata(self) -> Dict[int, SaveMetadata]:
 59        """Load save metadata from JSON file"""
 60        metadata_path = self._get_metadata_path()
 61        if not os.path.exists(metadata_path):
 62            return {}
 63            
 64        try:
 65            with open(metadata_path, 'r') as f:
 66                data = json.load(f)
 67                # Convert back to SaveMetadata objects
 68                metadata = {}
 69                for slot_str, meta_dict in data.items():
 70                    slot = int(slot_str)
 71                    metadata[slot] = SaveMetadata(
 72                        slot_number=slot,
 73                        save_name=meta_dict['save_name'],
 74                        timestamp=meta_dict['timestamp'],
 75                        turn_number=meta_dict['turn_number'],
 76                        current_player=meta_dict['current_player'],
 77                        knight_count=meta_dict['knight_count'],
 78                        vs_ai=meta_dict['vs_ai'],
 79                        board_size=tuple(meta_dict['board_size'])
 80                    )
 81                return metadata
 82        except Exception as e:
 83            print(f"Error loading metadata: {e}")
 84            return {}
 85            
 86    def _save_metadata(self):
 87        """Save metadata to JSON file"""
 88        try:
 89            # Convert SaveMetadata objects to dicts
 90            data = {}
 91            for slot, meta in self.metadata.items():
 92                data[str(slot)] = {
 93                    'save_name': meta.save_name,
 94                    'timestamp': meta.timestamp,
 95                    'turn_number': meta.turn_number,
 96                    'current_player': meta.current_player,
 97                    'knight_count': meta.knight_count,
 98                    'vs_ai': meta.vs_ai,
 99                    'board_size': list(meta.board_size)
100                }
101                
102            with open(self._get_metadata_path(), 'w') as f:
103                json.dump(data, f, indent=2)
104        except Exception as e:
105            print(f"Error saving metadata: {e}")
106            
107    def get_save_slots(self) -> List[Optional[SaveMetadata]]:
108        """Get list of save slots with their metadata (None for empty slots)"""
109        slots = []
110        for i in range(1, self.MAX_SLOTS + 1):
111            if i in self.metadata:
112                slots.append(self.metadata[i])
113            else:
114                slots.append(None)
115        return slots

117    def save_game(self, game_state, slot: int, save_name: Optional[str] = None) -> Dict[str, Any]:
118        """
119        Save game state to specified slot
120        
121        Args:
122            game_state: The GameState object to save
123            slot: Save slot number (1-10)
124            save_name: Optional custom save name
125            
126        Returns:
127            Dict with 'success' bool and 'message' string
128        """
129        if not 1 <= slot <= self.MAX_SLOTS:
130            return {'success': False, 'message': f'Invalid slot number: {slot}'}
131            
132        try:
133            # Create save data dictionary with all necessary state
134            save_data = {
135                'board_width': game_state.board_width,
136                'board_height': game_state.board_height,
137                'tile_size': game_state.tile_size,
138                'current_player': game_state.current_player,
139                'turn_number': game_state.turn_number,
140                'vs_ai': game_state.vs_ai,
141                'ai_difficulty': game_state.ai_player.difficulty if game_state.ai_player else None,
142                
143                # Units and castles
144                'knights': self._serialize_knights(game_state.knights),
145                'castles': self._serialize_castles(game_state.castles),
146                
147                # Terrain
148                'terrain_map': self._serialize_terrain(game_state.terrain_map),
149                
150                # Fog of war
151                'fog_of_war': self._serialize_fog_of_war(game_state.fog_of_war),
152                
153                # Camera position
154                'camera_x': game_state.camera_x,
155                'camera_y': game_state.camera_y,
156                
157                # Messages
158                'messages': game_state.messages,
159                
160                # Movement history
161                'movement_history': game_state.movement_history,
162            }
163            
164            # Save to file
165            filename = self._get_save_filename(slot)
166            with open(filename, 'wb') as f:
167                pickle.dump(save_data, f)
168                
169            # Update metadata
170            if save_name is None:
171                save_name = f"Turn {game_state.turn_number}"
172                
173            self.metadata[slot] = SaveMetadata(
174                slot_number=slot,
175                save_name=save_name,
176                timestamp=datetime.now().strftime("%Y-%m-%d %H:%M:%S"),
177                turn_number=game_state.turn_number,
178                current_player=game_state.current_player,
179                knight_count=len(game_state.knights),
180                vs_ai=game_state.vs_ai,
181                board_size=(game_state.board_width, game_state.board_height)
182            )
183            self._save_metadata()
184            
185            return {'success': True, 'message': f'Game saved to slot {slot}'}
186            
187        except Exception as e:
188            return {'success': False, 'message': f'Error saving game: {str(e)}'}
189            
190    def load_game(self, slot: int) -> Dict[str, Any]:
191        """
192        Load game state from specified slot
193        
194        Args:
195            slot: Save slot number (1-10)
196            
197        Returns:
198            Dict with 'success' bool, 'message' string, and 'data' dict if successful

200        if not 1 <= slot <= self.MAX_SLOTS:
201            return {'success': False, 'message': f'Invalid slot number: {slot}'}
202            
203        if slot not in self.metadata:
204            return {'success': False, 'message': f'No save in slot {slot}'}
205            
206        filename = self._get_save_filename(slot)
207        if not os.path.exists(filename):
208            return {'success': False, 'message': f'Save file missing for slot {slot}'}
209            
210        try:
211            with open(filename, 'rb') as f:
212                save_data = pickle.load(f)
213                
214            return {
215                'success': True, 
216                'message': f'Game loaded from slot {slot}',
217                'data': save_data
218            }
219            
220        except Exception as e:
221            return {'success': False, 'message': f'Error loading game: {str(e)}'}

267    def _serialize_knights(self, knights):
268        """Serialize knight data for saving"""
269        serialized = []
270        for knight in knights:
271            data = {
272                'name': knight.name,
273                'unit_class': knight.unit_class.value,
274                'x': knight.x,
275                'y': knight.y,
276                'player_id': knight.player_id,
277                'has_moved': knight.has_moved,
278                'has_acted': knight.has_acted,
279                'has_used_special': knight.has_used_special,
280                'max_action_points': knight.max_action_points,
281                'action_points': knight.action_points,
282                
283                # Stats
284                'current_soldiers': knight.stats.stats.current_soldiers,
285                'max_soldiers': knight.stats.stats.max_soldiers,
286                'morale': knight.stats.stats.morale,
287                'will': knight.stats.stats.will,
288                'max_will': knight.stats.stats.max_will,
289                
290                # State flags
291                'is_garrisoned': knight.is_garrisoned,
292                'is_disrupted': knight.is_disrupted,
293                'is_routing': knight.is_routing,
294                'in_enemy_zoc': knight.in_enemy_zoc,
295                'is_engaged_in_combat': knight.is_engaged_in_combat,
296                
297                # Facing
298                'facing': knight.facing.facing.value if hasattr(knight, 'facing') else None,
299                
300                # Generals
301                'generals': self._serialize_generals(knight.generals) if hasattr(knight, 'generals') else []
302            }
303            serialized.append(data)
304        return serialized
305        
306    def _serialize_generals(self, general_roster):
307        """Serialize general roster data"""
308        generals = []
309        for general in general_roster.generals:
310            # Serialize abilities by their class names
311            ability_names = []
312            for ability in general.abilities:
313                ability_names.append(ability.__class__.__name__)
314                
315            generals.append({
316                'name': general.name,
317                'title': general.title,
318                'level': general.level,
319                'experience': general.experience,
320                'abilities': ability_names
321            })
322        return generals
323        
324    def _serialize_castles(self, castles):
325        """Serialize castle data"""
326        serialized = []
327        for castle in castles:
328            data = {
329                'center_x': castle.center_x,
330                'center_y': castle.center_y,
331                'player_id': castle.player_id,
332                'max_health': castle.max_health,
333                'health': castle.health,
334                'defense': castle.defense,
335                'arrow_damage_per_archer': castle.arrow_damage_per_archer,
336                'arrow_range': castle.arrow_range,
337                'garrison_slots': castle.garrison_slots,
338                'garrisoned_units': [unit.name for unit in castle.garrisoned_units]
339            }
340            serialized.append(data)
341        return serialized
342        
343    def _serialize_terrain(self, terrain_map):
344        """Serialize terrain map data"""
345        if not terrain_map:
346            return None
347            
348        terrain_data = []
349        for x in range(terrain_map.width):
350            for y in range(terrain_map.height):
351                terrain = terrain_map.get_terrain(x, y)
352                if terrain:
353                    terrain_data.append({
354                        'x': x,
355                        'y': y,
356                        'type': terrain.type.value
357                    })
358        return terrain_data
359        
360    def _serialize_fog_of_war(self, fog_of_war):
361        """Serialize fog of war data"""
362        if not fog_of_war:
363            return None
364            
365        fog_data = {
366            'width': fog_of_war.width,
367            'height': fog_of_war.height,
368            'num_players': fog_of_war.num_players,
369            'visibility_maps': {}
370        }
371        
372        # Save visibility state for each player
373        for player_id, vis_map in fog_of_war.visibility_maps.items():
374            fog_data['visibility_maps'][player_id] = {}
375            for coords, state in vis_map.items():
376                # Convert tuple key to string for JSON serialization
377                key = f"{coords[0]},{coords[1]}"
378                fog_data['visibility_maps'][player_id][key] = state.value
379                
380        return fog_data
