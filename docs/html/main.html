204    def _handle_game(self, dt):
205        for event in pygame.event.get():
206            if event.type == pygame.QUIT:
207                self.running = False
208            elif event.type == pygame.KEYDOWN and event.key == pygame.K_ESCAPE:
209                # Toggle pause menu
210                self.paused = not self.paused
211                if self.paused:
212                    self.pause_menu.show()
213                else:
214                    self.pause_menu.hide()
215            elif self.paused:
216                # Handle save/load menu if visible
217                if self.save_load_menu.visible:
218                    result = self.save_load_menu.handle_event(event)
219                    if result:
220                        if result['action'] == 'save':
221                            # Save the game
222                            self.game_state.prepare_for_save()
223                            save_result = self.save_manager.save_game(
224                                self.game_state,
225                                result['slot'],
226                                result.get('name')
227                            )
228                            print(save_result['message'])
229                            if save_result['success']:
230                                self.save_load_menu.hide()
231                        elif result['action'] == 'load':
232                            # Load the game
233                            load_result = self.save_manager.load_game(result['slot'])
234                            if load_result['success']:
235                                # Connect renderer to game state for zoom consistency
236                                self.game_state.renderer = self.renderer
237                                self.game_state.restore_after_load(load_result['data'])
238                                self.save_load_menu.hide()
239                                self.paused = False
240                                self.pause_menu.hide()
241                                print(load_result['message'])
242                            else:
243                                print(f"Load failed: {load_result['message']}")
244                else:
245                    # Handle pause menu events
246                    option = self.pause_menu.handle_event(event)
247                    if option == MenuOption.RESUME:
248                        self.paused = False
249                        self.pause_menu.hide()
250                    elif option == MenuOption.NEW_GAME:
251                        self.paused = False
252                        self.in_game = False
253                        self.in_mode_select = True
254                        self.pause_menu.hide()
255                    elif option == MenuOption.SAVE_GAME:
256                        self.save_load_menu.show(SaveLoadAction.SAVE)
257                    elif option == MenuOption.LOAD_GAME:
258                        self.save_load_menu.show(SaveLoadAction.LOAD)
259                    elif option == MenuOption.OPTIONS:
260                        print("Options - Not implemented yet")
261                    elif option == MenuOption.QUIT:
262                        self.running = False
263            else:
264                self.input_handler.handle_event(event, self.game_state)
265        
266        if not self.paused:
267            self.game_state.update(dt)
268        
269        self.renderer.render(self.game_state)
270        
271        if self.paused:
272            self.pause_menu.draw()
273            
274        # Draw save/load menu on top if visible
275        if self.save_load_menu.visible:
276            self.save_load_menu.draw()
